79 std::shared_ptr<const T> bindWithPartId(std::shared_ptr<const T> obj, Cmper requestedPartId) const
80 {
81 if constexpr (std::is_base_of_v<OthersBase, T> || std::is_base_of_v<DetailsBase, T>) {

93 struct ObjectKey {

95 Cmper partId;
96 std::optional<Cmper> cmper1;

101 ObjectKey(std::string_view n,
102 Cmper p,
103 std::optional<Cmper> c1 = std::nullopt,
104 std::optional<Cmper> c2 = std::nullopt,
105 std::optional<Inci> i = std::nullopt) : nodeId(n), partId(p), cmper1(c1), cmper2(c2), inci(i)
106 {
107 }
108
110 inline bool operator<(const ObjectKey& other) const noexcept
111 {
112 if (nodeId != other.nodeId) return nodeId < other.nodeId;
113 if (partId != other.partId) return partId < other.partId;
114
115 // GCC O3 emits spurious warnings if we directly compare std::optional values,
116 // so this code does the comparisons explicitly.
117
118 // cmper1: disengaged < engaged; if both engaged, compare payloads
119 const bool a1 = cmper1.has_value(), b1 = other.cmper1.has_value();
120 if (a1 != b1) return !a1 && b1;
121 if (a1 && *cmper1 != *other.cmper1) return *cmper1 < *other.cmper1;
122
123 // cmper2
124 const bool a2 = cmper2.has_value(), b2 = other.cmper2.has_value();
125 if (a2 != b2) return !a2 && b2;
126 if (a2 && *cmper2 != *other.cmper2) return *cmper2 < *other.cmper2;
127
128 // inci
129 const bool ai = inci.has_value(), bi = other.inci.has_value();
130 if (ai != bi) return !ai && bi;
131 if (ai) return *inci < *other.inci;
132
133 return false;
134 }
135
137 std::string description() const
138 {
139 std::string result = std::string(nodeId) + " part " + std::to_string(partId);
140 if (cmper1) {
141 if (cmper2) {
142 result += " cmpers [" + std::to_string(cmper1.value()) + ", " + std::to_string(cmper2.value()) + "]";
143 } else {
144 result += " cmper " + std::to_string(cmper1.value());
145 }
146 }
147 if (inci) {
148 result += " inci " + std::to_string(inci.value());
149 }
150 return result;
151 }

163 void add(ObjectKey key, ObjectPtr object)
164 {
165 if (key.inci.has_value()) {
166 ObjectKey noInciKey = key;
167 noInciKey.inci = std::nullopt;
168 auto currentIncis = getArray<ObjectBaseType>(noInciKey, key.partId);
169 if (key.inci.value() != int(currentIncis.size())) {
170 MUSX_INTEGRITY_ERROR("Node " + std::string(key.nodeId) + " has inci " + std::to_string(key.inci.value()) + " that is out of sequence.");
171 }
172 }
173 auto shareModeIt = m_shareMode.find(key.nodeId);
174 auto [poolIt, emplaced] = m_pool.emplace(std::move(key), object);
175 if (!emplaced) {
176 MUSX_INTEGRITY_ERROR("Attempted to add same key more than once: " + poolIt->first.description());
177 }
178 if (shareModeIt == m_shareMode.end()) {
179 m_shareMode.emplace(poolIt->first.nodeId, object->getShareMode());
180 } else if (object->getShareMode() != shareModeIt->second && object->getShareMode() != Base::ShareMode::All) {
181 if (shareModeIt->second == Base::ShareMode::All) {
182 m_shareMode[poolIt->first.nodeId] = object->getShareMode();
183 } else {
184 MUSX_INTEGRITY_ERROR("Share mode for added " + std::string(poolIt->first.nodeId) + " object [" + std::to_string(int(object->getShareMode()))
185 + "] does not match previous [" + std::to_string(int(shareModeIt->second)) + "]");
186 }
187 }
188 }

203 template <typename T>
204 MusxInstanceList<T> getArray(const ObjectKey& key, Cmper requestedPartId) const
205 {
206 MusxInstanceList<T> result(m_document, requestedPartId);
207
208 auto rangeStart = m_pool.lower_bound(key);
209 auto rangeEnd = m_pool.upper_bound(
210 ObjectKey{
211 key.nodeId,
212 key.partId,
213 key.cmper1.value_or((std::numeric_limits<Cmper>::max)()),
214 key.cmper2.value_or((std::numeric_limits<Cmper>::max)()),
215 key.inci.value_or((std::numeric_limits<Inci>::max)())
216 }
217 );
218
219 for (auto it = rangeStart; it != rangeEnd; ++it) {
220 auto typedPtr = bindWithPartId<T>(std::dynamic_pointer_cast<T>(it->second), requestedPartId);
221 assert(typedPtr);
222 result.push_back(typedPtr);
223 }
224 return result;
225 }

239 template <typename T>
240 MusxInstanceList<T> getArrayForPart(const ObjectKey& key) const
241 {
242 Base::ShareMode forShareMode = Base::ShareMode::All;
243 if (key.partId != SCORE_PARTID) {
244 auto it = m_shareMode.find(key.nodeId);
245 // If the nodeId is not found in m_shareMode, it means the document contains no instances,
246 // and every path will return an empty vector. We can safely ignore this case.
247 if (it != m_shareMode.end()) {
248 forShareMode = it->second;
249 }
250 if (forShareMode == Base::ShareMode::Partial) {
251 if constexpr (std::is_base_of_v<OthersBase, T>) {
252 if (!key.cmper1.has_value()) {
253 throw std::invalid_argument("Array searches on partially shared Others must supply a cmper.");
254 }
255 } else if constexpr (std::is_base_of_v<DetailsBase, T>) {
256 if (!key.cmper1.has_value() || !key.cmper2.has_value()) {
257 throw std::invalid_argument("Array searches on partially shared Details must supply both cmpers.");
258 }
259 }
260 }
261 }
262 auto keyResult = getArray<T>(key, key.partId);
263 if (!keyResult.empty() || key.partId == SCORE_PARTID || forShareMode == Base::ShareMode::None) {
264 return keyResult;
265 }
266 ObjectKey scoreKey(key);
267 scoreKey.partId = SCORE_PARTID;
268 return getArray<T>(scoreKey, key.partId);
269 }

282 template <typename T>
283 MusxInstance<T> getSource(const ObjectKey& key) const
284 {
285 auto it = m_pool.find(key);
286 if (it == m_pool.end()) {
287 return nullptr;
288 }
289 auto typedPtr = std::dynamic_pointer_cast<T>(it->second);
290 assert(typedPtr); // There is a program bug if the pointer cast fails.
291 return typedPtr;
292 }

307 template <typename T>
308 MusxInstance<T> getEffectiveSourceForPart(const ObjectKey& key) const
309 {
310 if (auto partVersion = getSource<T>(key)) {
311 return partVersion;
312 }
313 if (key.partId == SCORE_PARTID) {
314 // if this is already the score version, there is nothing to return.
315 return nullptr;
316 } else {
317 // if this is for a part, do not search score for unshared types.
318 auto it = m_shareMode.find(key.nodeId);
319 if (it != m_shareMode.end() && it->second == Base::ShareMode::None) {
320 return nullptr;
321 }
322 }
323 ObjectKey scoreKey(key);
324 scoreKey.partId = SCORE_PARTID;
325 return getSource<T>(scoreKey);
326 }

341 template <typename T>
342 MusxInstance<T> getEffectiveForPart(const ObjectKey& key) const
343 {
344 return bindWithPartId<T>(getEffectiveSourceForPart<T>(key), key.partId);
345 }
