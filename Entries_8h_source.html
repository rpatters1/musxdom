513class NoteInfoPtr
514{
515public:
517 NoteInfoPtr() : m_entry(), m_noteIndex(0) {}
518
522 NoteInfoPtr(const EntryInfoPtr& entryInfo, size_t noteIndex)
523 : m_entry(entryInfo), m_noteIndex(noteIndex)
524 {}
525
527 operator bool() const
528 { return m_entry && m_noteIndex < m_entry->getEntry()->notes.size(); }
529
531 bool isSameNote(const NoteInfoPtr& src) const
532 { return m_entry.isSameEntry(src.m_entry) && m_noteIndex == src.m_noteIndex; }

535 std::shared_ptr<const Note> operator->() const
536 {
537 MUSX_ASSERT_IF(m_noteIndex >= m_entry->getEntry()->notes.size()) {
538 throw std::logic_error("Note index is too large for notes array.");
539 }
540 return m_entry->getEntry()->notes[m_noteIndex];
