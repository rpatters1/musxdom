633class NoteInfoPtr
634{
635public:
637 NoteInfoPtr() : m_entry(), m_noteIndex(0) {}
638
642 NoteInfoPtr(const EntryInfoPtr& entryInfo, size_t noteIndex)
643 : m_entry(entryInfo), m_noteIndex(noteIndex)
644 {}
645
647 operator bool() const
648 { return m_entry && m_noteIndex < m_entry->getEntry()->notes.size(); }
649
651 bool isSameNote(const NoteInfoPtr& src) const
652 { return m_entry.isSameEntry(src.m_entry) && m_noteIndex == src.m_noteIndex; }
653
657 NoteInfoPtr findEqualPitch(const EntryInfoPtr& entry) const;
658
660 std::shared_ptr<const Note> operator->() const
661 {
662 MUSX_ASSERT_IF(m_noteIndex >= m_entry->getEntry()->notes.size()) {
663 throw std::logic_error("Note index is too large for notes array.");
664 }
665 return m_entry->getEntry()->notes[m_noteIndex];
666 }
667
669 EntryInfoPtr getEntryInfo() const
670 { return m_entry; }
671
682 std::tuple<Note::NoteName, int, int, int> calcNoteProperties(const std::optional<bool>& enharmonicRespell = std::nullopt) const;
683
687 NoteInfoPtr calcTieTo() const;
688
692 NoteInfoPtr calcTieFrom() const;
693
695 InstCmper calcStaff() const;
696
699 std::unique_ptr<music_theory::Transposer> createTransposer() const;
700
703 NoteInfoPtr getNext() const
704 {
705 if (m_noteIndex >= m_entry->getEntry()->notes.size()) {
706 return NoteInfoPtr();
707 }
708 return NoteInfoPtr(m_entry, m_noteIndex + 1);
709 }
710
713 NoteInfoPtr getPrevious() const
714 {
715 if (m_noteIndex <= 0) {
716 return NoteInfoPtr();
717 }
718 return NoteInfoPtr(m_entry, m_noteIndex - 1);
719 }
720
721private:
725 bool isSamePitchValues(const NoteInfoPtr& src) const
726 {
727 if (!*this || !src) {
728 return false;
729 }
730 return (*this)->harmLev == src->harmLev
731 && (*this)->harmAlt == src->harmAlt;
732 }
733
734 EntryInfoPtr m_entry;
735 size_t m_noteIndex;
