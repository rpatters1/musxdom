549class NoteInfoPtr
550{
551public:
553 NoteInfoPtr() : m_entry(), m_noteIndex(0) {}
554
558 NoteInfoPtr(const EntryInfoPtr& entryInfo, size_t noteIndex)
559 : m_entry(entryInfo), m_noteIndex(noteIndex)
560 {}
561
563 operator bool() const
564 { return m_entry && m_noteIndex < m_entry->getEntry()->notes.size(); }
565
567 bool isSameNote(const NoteInfoPtr& src) const
568 { return m_entry.isSameEntry(src.m_entry) && m_noteIndex == src.m_noteIndex; }
569
573 NoteInfoPtr findEqualPitch(const EntryInfoPtr& entry) const;
574
576 std::shared_ptr<const Note> operator->() const
577 {
578 MUSX_ASSERT_IF(m_noteIndex >= m_entry->getEntry()->notes.size()) {
579 throw std::logic_error("Note index is too large for notes array.");
580 }
581 return m_entry->getEntry()->notes[m_noteIndex];
582 }
583
585 EntryInfoPtr getEntryInfo() const
586 { return m_entry; }
587
598 std::tuple<Note::NoteName, int, int, int> calcNoteProperties(const std::optional<bool>& enharmonicRespell = std::nullopt) const;
599
603 NoteInfoPtr calcTieTo() const;
604
608 NoteInfoPtr calcTieFrom() const;
609
611 InstCmper calcStaff() const;
612
615 std::unique_ptr<music_theory::Transposer> createTransposer() const;
616
619 NoteInfoPtr getNext() const
620 {
621 if (m_noteIndex >= m_entry->getEntry()->notes.size()) {
622 return NoteInfoPtr();
623 }
624 return NoteInfoPtr(m_entry, m_noteIndex + 1);
625 }
626
629 NoteInfoPtr getPrevious() const
630 {
631 if (m_noteIndex <= 0) {
632 return NoteInfoPtr();
633 }
634 return NoteInfoPtr(m_entry, m_noteIndex - 1);
635 }
636
637private:
641 bool isSamePitchValues(const NoteInfoPtr& src) const
642 {
643 if (!*this || !src) {
644 return false;
645 }
646 return (*this)->harmLev == src->harmLev
647 && (*this)->harmAlt == src->harmAlt;
648 }
649
650 EntryInfoPtr m_entry;
651 size_t m_noteIndex;
652};
