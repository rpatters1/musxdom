41inline StaffGroup::BracketStyle toEnum<StaffGroup::BracketStyle>(const int& value)
42{
43 if (value >= static_cast<int>(StaffGroup::BracketStyle::None) &&
44 value <= static_cast<int>(StaffGroup::BracketStyle::DeskBracket)) {
45 return static_cast<StaffGroup::BracketStyle>(value);
46 }
47 MUSX_UNKNOWN_XML("Invalid <bracket><id> value in XML for StaffGroup: " + std::to_string(value));

51MUSX_RESOLVER_ENTRY(StaffGroup, {
52 [](const dom::DocumentPtr& document) {
53 auto parts = document->getOthers()->getArray<others::PartDefinition>(SCORE_PARTID);
54 for (const auto& part : parts) {
55 auto groups = document->getDetails()->getArray<StaffGroup>(part->getCmper(), BASE_SYSTEM_ID);
56 auto baseList = document->getOthers()->getArray<others::InstrumentUsed>(part->getCmper(), BASE_SYSTEM_ID);
57 for (const auto& instance : groups) {
58 auto startIndex = others::InstrumentUsed::getIndexForStaff(baseList, instance->startInst);
59 auto endIndex = others::InstrumentUsed::getIndexForStaff(baseList, instance->endInst);
60 if (!startIndex || !endIndex) {
61 // this situation arises fairly commonly in Finale files, so the message is demoted to a verbose logging message
62 // from an integrity error.
63 util::Logger::log(util::Logger::LogLevel::Verbose, "Group " + std::to_string(instance->getCmper2()) + " in part " + part->getName()
64 + " [" + std::to_string(part->getCmper()) + "] has non-existent start or end staff cmpers");
65 continue;
66 }
67 for (size_t x = *startIndex; x <= *endIndex && x < baseList.size(); x++) {
68 instance->staves.insert(baseList[x]->staffId);
69 }
70 }
71 }
72 }
73});
