68class ElementLinker {
69public:
77 using Resolver = std::function<void(const dom::DocumentPtr&)>;
78
88 void addResolver(Resolver resolver, const std::string_view& key)
89 {
90 assert(!key.empty());
91 if (resolvers.count(std::string(key)) > 0) {
92 return; // Skip if already registered
93 }
94
95 resolvers[std::string(key)] = std::move(resolver);
96 }
97
106 void resolveAll(const dom::DocumentPtr& document)
107 {
108 for (const auto& [key, resolver] : resolvers) {
109 resolver(document);
110 }
111
112 resolvers.clear();
113 }
114
115private:
122 std::map<std::string, Resolver> resolvers;
123};

266template <typename T>
267struct FieldPopulator : public FactoryBase
268{
269 static void populateField(const std::shared_ptr<T>& instance, const XmlElementPtr& fieldElement)
270 {
271 auto it = elementXref().find(fieldElement->getTagName());
272 if (it != elementXref().end()) {
273 std::get<1>(*it)(fieldElement, instance);
274 } else {
275 const bool requireFields = [instance]() {
276 if constexpr (std::is_base_of_v<Base, T>) {
277 return instance->requireAllFields();
278 } else {
279 return true;
280 }
281 }();
282 if (requireFields) {
283 MUSX_UNKNOWN_XML("xml element <" + fieldElement->getParent()->getTagName() + "> has child <" + fieldElement->getTagName() + "> which is not in the element list.");
284 }
285 }
286 }
287
288 static void populate(const std::shared_ptr<T>& instance, const XmlElementPtr& element)
289 {
290 for (auto child = element->getFirstChildElement(); child; child = child->getNextSibling()) {
291 populateField(instance, child);
292 }
293 if constexpr (std::is_base_of_v<Base, T>) {
294 instance->integrityCheck();
295 }
296 }
297
298 static void populate(const std::shared_ptr<T>& instance, const XmlElementPtr& element, ElementLinker& elementLinker)
299 {
300 populate(instance, element);
301 if (ResolverContainer<T>::resolver.has_value()) {
302 elementLinker.addResolver(ResolverContainer<T>::resolver.value(), element->getTagName());
303 }
304 }
305
306 template <typename... Args>
307 static std::shared_ptr<T> createAndPopulate(const XmlElementPtr& element, Args&&... args)
308 {
309 return FieldPopulator<T>::createAndPopulateImpl(element, std::forward<Args>(args)...);
310 }
311
312private:
313 static const std::unordered_map<std::string_view, XmlElementPopulator<T>>& elementXref()
314 {
315 static const std::unordered_map<std::string_view, XmlElementPopulator<T>> xref = []()
316 {
317 std::unordered_map<std::string_view, XmlElementPopulator<T>> retval;
318 auto mappingArray = T::xmlMappingArray();
319 for (std::size_t i = 0; i < mappingArray.size(); i++) {
320 const XmlElementDescriptor<T> descriptor = mappingArray[i];
321 retval[std::get<0>(descriptor)] = std::get<1>(descriptor);
322 }
323 return retval;
324 }();
325 return xref;
326 }
327
328 template <typename... Args>
329 static std::shared_ptr<T> createAndPopulateImpl(const XmlElementPtr& element, Args&&... args)
330 {
331 auto instance = std::make_shared<T>(std::forward<Args>(args)...);
332 FieldPopulator<T>::populate(instance, element);
333 return instance;
334 }
335};

342template <typename EnumClass, typename EmbeddedClass>
343static void populateEmbeddedClass(const XmlElementPtr& e, std::unordered_map<EnumClass, std::shared_ptr<EmbeddedClass>>& listArray)
344{
345 auto typeAttr = e->findAttribute("type");
346 if (!typeAttr) {
347 throw std::invalid_argument("<" + e->getTagName() + "> element has no type attribute");
348 }
349 listArray.emplace(toEnum<EnumClass>(typeAttr->getValueTrimmed()), FieldPopulator<EmbeddedClass>::createAndPopulate(e));
350}

357template <typename T>
358static std::vector<T> populateEmbeddedArray(const XmlElementPtr& e, const std::string_view& elementNodeName)
359{
360 std::vector<T> result;
361 for (auto child = e->getFirstChildElement(); child; child = child->getNextSibling()) {
362 if (child->getTagName() != elementNodeName) {
363 MUSX_UNKNOWN_XML("Unknown tag <" + child->getTagName() + "> while processing embedded xml array <" + e->getTagName() + ">");
364 continue;
365 }
366 if constexpr (std::is_fundamental_v<T>) {
367 result.push_back(child->getTextAs<T>());
368 } else if constexpr (std::is_same_v<T, std::string>) {
369 result.push_back(child->getText());
370 } else {
371 result.push_back(FieldPopulator<T>::createAndPopulate(child));
372 }
373 }
374 return result;
375}
