113constexpr int calcDisplacement(int pitchClass, int octave)
114{
115 pitchClass %= STANDARD_DIATONIC_STEPS;
116 const int relativeOctave = octave - 4;
117
118 return pitchClass + (STANDARD_DIATONIC_STEPS * relativeOctave);
119}

124template <typename T>
125constexpr T sign(T n)
126{
127 static_assert(std::is_arithmetic_v<T>, "sign requires a numeric type");
128 return n < T(0) ? T(-1) : T(1);
129}
130
136template <typename T>
137constexpr T signedModulus(T n, T d)
138{
139 static_assert(std::is_integral_v<T>, "signedModulus requires an integer type");
140 return sign(n) * (std::abs(n) % d);
141}
142
149template <typename T>
150constexpr T positiveModulus(T n, T d, T* q = nullptr)
151{
152 static_assert(std::is_integral_v<T>, "positiveModulus requires an integer type");
153 if (q) *q = n / d;
154 T result = signedModulus(n, d);
155 if (result < 0) {
156 result += d;
157 if (q) --(*q);
158 }
159 return result;
160}

166constexpr int calc12EdoHalfstepsInInterval(int interval, int chromaticAlteration)
167{
168 int octaves{};
169 int diatonic = positiveModulus(interval, STANDARD_DIATONIC_STEPS, &octaves);
170 return MAJOR_KEYMAP[diatonic] + (octaves * STANDARD_12EDO_STEPS) + chromaticAlteration;
171}
172
177constexpr int calcAlterationFrom12EdoHalfsteps(int interval, int halfsteps)
178{
179 int octaves{};
180 int diatonic = positiveModulus(interval, STANDARD_DIATONIC_STEPS, &octaves);
181 int expectedHalfsteps = MAJOR_KEYMAP[diatonic] + (octaves * STANDARD_12EDO_STEPS);
182 return halfsteps - expectedHalfsteps;
183}

189constexpr int calcAlterationFromKeySigChange(int interval, int keySigChange)
190{
191 int diatonic = positiveModulus(interval, STANDARD_DIATONIC_STEPS);
192 int expectedKeyChange = DIATONIC_INTERVAL_ADJUSTMENTS[diatonic][0];
193 if (interval < 0) {
194 if (std::abs(expectedKeyChange) > 1) { // imperfect intervals
195 expectedKeyChange -= STANDARD_DIATONIC_STEPS;
196 }
197 }
198 int alteration = (keySigChange - expectedKeyChange) / STANDARD_DIATONIC_STEPS;
199 return alteration;
200}

205constexpr bool calcTranspositionIsOctave(int displacement, int alteration)
206{
207 return (displacement % STANDARD_DIATONIC_STEPS) == 0 && alteration == 0;

217class Transposer
218{
219private:
220 int m_displacement;
221 int m_alteration; // alteration from key signature
222 int m_numberOfEdoDivisions; // number of divisions in the EDO (default 12)
223 std::vector<int> m_keyMap; // step map for the EDO
224
225public:
234 Transposer(int displacement, int alteration,
235 bool isMinor = false, int numberOfEdoDivisions = STANDARD_12EDO_STEPS,
236 const std::optional <std::vector<int>>& keyMap = std::nullopt)
237 : m_displacement(displacement), m_alteration(alteration), m_numberOfEdoDivisions(numberOfEdoDivisions)
238 {
239 if (keyMap) {
240 if (keyMap.value().size() != STANDARD_DIATONIC_STEPS) {
241 throw std::invalid_argument("The Transposer class only supports key map arrays of " + std::to_string(STANDARD_DIATONIC_STEPS) + " elements");
242 }
243 m_keyMap = keyMap.value();
244 } else if (isMinor) {
245 m_keyMap.assign(MINOR_KEYMAP.begin(), MINOR_KEYMAP.end());
246 } else {
247 m_keyMap.assign(MAJOR_KEYMAP.begin(), MAJOR_KEYMAP.end());
248 }
249 }
250
252 int displacement() const { return m_displacement; }
253
255 int alteration() const { return m_alteration; }
256
259 void diatonicTranspose(int interval)
260 {
261 m_displacement += interval;
262 }
263
266 void enharmonicTranspose(int direction)
267 {
268 const int keyStepEnharmonic = calcStepsBetweenScaleDegrees(m_displacement, m_displacement + sign(direction));
269 diatonicTranspose(sign(direction));
270 m_alteration -= sign(direction) * keyStepEnharmonic;
271 }

294 void chromaticTranspose(int interval, int chromaticAlteration)
295 {
296 const int intervalNormalized = signedModulus(interval, STANDARD_DIATONIC_STEPS);
297 const int stepsInAlteration = calcStepsInAlteration(interval, chromaticAlteration);
298 const int stepsInInterval = calcStepsInNormalizedInterval(intervalNormalized);
299 const int stepsInDiatonicInterval = calcStepsBetweenScaleDegrees(m_displacement, m_displacement + intervalNormalized);
300
301 const int effectiveAlteration = stepsInAlteration + stepsInInterval - sign(interval) * stepsInDiatonicInterval;
302
303 diatonicTranspose(interval);
304 m_alteration += effectiveAlteration;
305 }

313 void simplifySpelling()
314 {
315 while (std::abs(m_alteration) > 0) {
316 const int currSign = sign(m_alteration);
317 const int currAbsDisp = std::abs(m_alteration);
318 enharmonicTranspose(currSign);
319 if (std::abs(m_alteration) >= currAbsDisp) {
320 enharmonicTranspose(-currSign);
321 return;
322 }
323 if (currSign != sign(m_alteration)) {
324 break;
325 }
326 }
327 }

338 void stepwiseTranspose(int numberOfEdoDivisions)
339 {

341 simplifySpelling();

354 bool isEnharmonicEquivalent(int displacement, int alteration) const {
355 return calcAbsoluteDivision(displacement, alteration) == calcAbsoluteDivision(m_displacement, m_alteration);
356 }
357
358private:
359 int calcFifthSteps() const
360 {
361 // std::log(3.0 / 2.0) / std::log(2.0) is 0.5849625007211562.
362 static constexpr double kFifthsMultiplier = 0.5849625007211562;
363 return static_cast<int>(std::floor(m_numberOfEdoDivisions * kFifthsMultiplier) + 0.5);
364 }
365
366 int calcScaleDegree(int interval) const
367 { return positiveModulus(interval, int(m_keyMap.size())); }
368
369 int calcStepsBetweenScaleDegrees(int firstDisplacement, int secondDisplacement) const
370 {
371 const int firstScaleDegree = calcScaleDegree(firstDisplacement);
372 const int secondScaleDegree = calcScaleDegree(secondDisplacement);
373 int result = sign(secondDisplacement - firstDisplacement) * (m_keyMap[secondScaleDegree] - m_keyMap[firstScaleDegree]);
374 if (result < 0) {
375 result += m_numberOfEdoDivisions;
376 }
377 return result;
378 }
379
380 int calcStepsInAlteration(int interval, int alteration) const
381 {
382 const int fifthSteps = calcFifthSteps();
383 const int plusFifths = sign(interval) * alteration * 7; // number of fifths to add for a chromatic halfstep alteration (in any EDO)
384 const int minusOctaves = sign(interval) * alteration * -4; // number of octaves to subtract for a chromatic halfstep alteration (in any EDO)
385 const int result = sign(interval) * ((plusFifths * fifthSteps) + (minusOctaves * m_numberOfEdoDivisions));
386 return result;
387 }
388
389 int calcStepsInNormalizedInterval(int intervalNormalized) const
390 {
391 const int fifthSteps = calcFifthSteps();
392 const int index = std::abs(intervalNormalized);
393 const int plusFifths = DIATONIC_INTERVAL_ADJUSTMENTS[index][0]; // number of fifths
394 const int minusOctaves = DIATONIC_INTERVAL_ADJUSTMENTS[index][1]; // number of octaves
395
396 return sign(intervalNormalized) * ((plusFifths * fifthSteps) + (minusOctaves * m_numberOfEdoDivisions));
397 }
398
399 int calcAbsoluteDivision(int displacement, int alteration) const {
400 const int scaleDegree = calcScaleDegree(displacement); // 0..6
401 const int baseStep = m_keyMap[scaleDegree];
402
403 const int octaveCount = (displacement < 0 && displacement % STANDARD_DIATONIC_STEPS != 0)
404 ? (displacement / STANDARD_DIATONIC_STEPS) - 1
405 : (displacement / STANDARD_DIATONIC_STEPS);
406 const int octaveSteps = octaveCount * m_numberOfEdoDivisions;
407 const int chromaticSteps = calcStepsInAlteration(/*interval=*/+1, alteration);
408
409 return baseStep + chromaticSteps + octaveSteps;
410 }
411};
