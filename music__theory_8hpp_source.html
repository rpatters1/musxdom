45inline int calcDisplacement(int pitchClass, int octave)
46{
47 pitchClass %= STANDARD_DIATONIC_STEPS;
48 const int relativeOctave = octave - 4;
49
50 return pitchClass + (STANDARD_DIATONIC_STEPS * relativeOctave);
51}

56template <typename T>
57constexpr inline T sign(T n)
58{
59 static_assert(std::is_arithmetic_v<T>, "sign requires a numeric type");
60 return n < T(0) ? T(-1) : T(1);
61}
62
63template <typename T>
64constexpr T signedModulus(T n, T d) {
65 static_assert(std::is_integral_v<T>, "signed_modulus requires an integer type");
66 return sign(n) * (std::abs(n) % d);
67}

74class Transposer
75{
76private:
77 static constexpr std::array<int, STANDARD_DIATONIC_STEPS> MAJOR_KEYMAP = { 0, 2, 4, 5, 7, 9, 11 };
78 static constexpr std::array<int, STANDARD_DIATONIC_STEPS> MINOR_KEYMAP = { 0, 2, 3, 5, 7, 8, 10 };
79 int m_displacement;
80 int m_alteration;
81 int m_numberOfSteps; // number of steps in the EDO (default 12)
82 std::vector<int> m_keyMap; // step map for the EDO
83
84 static constexpr std::array<std::array<int, 2>, 7> DIATONIC_INTERVAL_ADJUSTMENTS = { {
85 { 0, 0}, // unison
86 { 2, -1}, // second
87 { 4, -2}, // third
88 {-1, 1}, // fourth
89 { 1, 0}, // fifth
90 { 3, -1}, // sixth
91 { 5, -2} // seventh
92 }};

94public:
101 Transposer(int displacement, int alteration,
102 bool isMinor = false, int numberOfSteps = STANDARD_12EDO_STEPS,
103 const std::optional <std::vector<int>>& keyMap = std::nullopt)
104 : m_displacement(displacement), m_alteration(alteration), m_numberOfSteps(numberOfSteps)
105 {
106 if (keyMap) {
107 if (keyMap.value().size() != STANDARD_DIATONIC_STEPS) {
108 throw std::invalid_argument("The Transposer class only supports key map arrays of " + std::to_string(STANDARD_DIATONIC_STEPS) + " elements");
109 }
110 m_keyMap = keyMap.value();
111 } else if (isMinor) {
112 m_keyMap.assign(MINOR_KEYMAP.begin(), MINOR_KEYMAP.end());
113 } else {
114 m_keyMap.assign(MAJOR_KEYMAP.begin(), MAJOR_KEYMAP.end());
115 }
116 }

126 void diatonicTranspose(int interval)
127 {
128 m_displacement += interval;
129 }

133 void enharmonicTranspose(int direction)
134 {
135 const int keyStepEnharmonic = calcStepsBetweenScaleDegrees(m_displacement, m_displacement + sign(direction));
136 diatonicTranspose(sign(direction));
137 m_alteration -= sign(direction) * keyStepEnharmonic;
138 }

158 void chromaticTranspose(int interval, int alteration)
159 {
160 const int intervalNormalized = signedModulus(interval, STANDARD_DIATONIC_STEPS);
161 const int stepsInAlteration = calcStepsInAlteration(interval, alteration);
162 const int stepsInInterval = calcStepsInNormalizedInterval(intervalNormalized);
163 const int stepsInDiatonicInterval = calcStepsBetweenScaleDegrees(m_displacement, m_displacement + intervalNormalized);
164
165 const int effectiveAlteration = stepsInAlteration + stepsInInterval - sign(interval) * stepsInDiatonicInterval;
166
167 diatonicTranspose(interval);
168 m_alteration += effectiveAlteration;
169 }

177 void simplifySpelling()
178 {
179 while (std::abs(m_alteration) > 0) {
180 const int currSign = sign(m_alteration);
181 const int currAbsDisp = std::abs(m_alteration);
182 enharmonicTranspose(currSign);
183 if (std::abs(m_alteration) >= currAbsDisp) {
184 enharmonicTranspose(-currSign);
185 return;
186 }
187 if (currSign != sign(m_alteration)) {
188 break;
189 }
190 }
191 }

203 void stepwiseTranspose(int numberOfSteps)
204 {
205 m_alteration += numberOfSteps;

219 bool isEnharmonicEquivalent(int displacement, int alteration) const {
220 return calcAbsoluteStep(displacement, alteration) == calcAbsoluteStep(m_displacement, m_alteration);
221 }
222
223private:
224 int calcFifthSteps() const
225 {
226 // std::log(3.0 / 2.0) / std::log(2.0) is 0.5849625007211562.
227 static constexpr double kFifthsMultiplier = 0.5849625007211562;
228 return static_cast<int>(std::floor(m_numberOfSteps * kFifthsMultiplier) + 0.5);
229 }
230
231 int calcScaleDegree(int interval) const
232 {
233 int intervalNormalized = signedModulus(interval, int(m_keyMap.size()));
234 if (intervalNormalized < 0) {
235 intervalNormalized += int(m_keyMap.size());
236 }
237 return intervalNormalized;
238 }
239
240 int calcStepsBetweenScaleDegrees(int firstDisplacement, int secondDisplacement) const
241 {
242 const int firstScaleDegree = calcScaleDegree(firstDisplacement);
243 const int secondScaleDegree = calcScaleDegree(secondDisplacement);
244 int result = sign(secondDisplacement - firstDisplacement) * (m_keyMap[secondScaleDegree] - m_keyMap[firstScaleDegree]);
245 if (result < 0) {
246 result += m_numberOfSteps;
247 }
248 return result;
249 }
250
251 int calcStepsInAlteration(int interval, int alteration) const
252 {
253 const int fifthSteps = calcFifthSteps();
254 const int plusFifths = sign(interval) * alteration * 7; // number of fifths to add for a chromatic halfstep alteration (in any EDO)
255 const int minusOctaves = sign(interval) * alteration * -4; // number of octaves to subtract for a chromatic halfstep alteration (in any EDO)
256 const int result = sign(interval) * ((plusFifths * fifthSteps) + (minusOctaves * m_numberOfSteps));
257 return result;
258 }
259
260 int calcStepsInNormalizedInterval(int intervalNormalized) const
261 {
262 const int fifthSteps = calcFifthSteps();
263 const int index = std::abs(intervalNormalized);
264 const int plusFifths = DIATONIC_INTERVAL_ADJUSTMENTS[index][0]; // number of fifths
265 const int minusOctaves = DIATONIC_INTERVAL_ADJUSTMENTS[index][1]; // number of octaves
266
267 return sign(intervalNormalized) * ((plusFifths * fifthSteps) + (minusOctaves * m_numberOfSteps));
268 }
269
270 int calcAbsoluteStep(int displacement, int alteration) const {
271 const int scaleDegree = calcScaleDegree(displacement); // 0..6
272 const int baseStep = m_keyMap[scaleDegree];
273
274 const int octaveCount = (displacement < 0 && displacement % STANDARD_DIATONIC_STEPS != 0)
275 ? (displacement / STANDARD_DIATONIC_STEPS) - 1
276 : (displacement / STANDARD_DIATONIC_STEPS);
277 const int octaveSteps = octaveCount * m_numberOfSteps;
278 const int chromaticSteps = calcStepsInAlteration(/*interval=*/+1, alteration);
279
280 return baseStep + chromaticSteps + octaveSteps;
281 }
