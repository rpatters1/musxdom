45constexpr int STANDARD_12EDO_STEPS = 12;

110inline int calcDisplacement(int pitchClass, int octave)
111{
112 pitchClass %= STANDARD_DIATONIC_STEPS;
113 const int relativeOctave = octave - 4;
114
115 return pitchClass + (STANDARD_DIATONIC_STEPS * relativeOctave);
116}

121template <typename T>
122constexpr inline T sign(T n)
123{
124 static_assert(std::is_arithmetic_v<T>, "sign requires a numeric type");
125 return n < T(0) ? T(-1) : T(1);
126}
127
133template <typename T>
134constexpr T signedModulus(T n, T d)
135{
136 static_assert(std::is_integral_v<T>, "signedModulus requires an integer type");
137 return sign(n) * (std::abs(n) % d);
138}
139
146template <typename T>
147constexpr T positiveModulus(T n, T d, T* q = nullptr)
148{
149 static_assert(std::is_integral_v<T>, "positiveModulus requires an integer type");
150 if (q) *q = n / d;
151 T result = signedModulus(n, d);
152 if (result < 0) {
153 result += d;
154 if (q) --(*q);
155 }
156 return result;
157}

163inline int calc12EdoHalfstepsInInterval(int interval, int chromaticAlteration)
164{
165 int octaves{};
166 int diatonic = positiveModulus(interval, STANDARD_DIATONIC_STEPS, &octaves);
167 return MAJOR_KEYMAP[diatonic] + (octaves * STANDARD_12EDO_STEPS) + chromaticAlteration;
168}
169
174inline int calcAlterationFrom12EdoHalfsteps(int interval, int halfsteps)
175{
176 int octaves{};
177 int diatonic = positiveModulus(interval, STANDARD_DIATONIC_STEPS, &octaves);
178 int expectedHalfsteps = MAJOR_KEYMAP[diatonic] + (octaves * STANDARD_12EDO_STEPS);
179 return halfsteps - expectedHalfsteps;
180}
181
186inline int calcAlterationFromKeySigChange(int interval, int keySigChange)
187{
188 int diatonic = positiveModulus(interval, STANDARD_DIATONIC_STEPS);
189 int expectedKeyChange = DIATONIC_INTERVAL_ADJUSTMENTS[diatonic][0];
190 if (interval < 0) {
191 if (std::abs(expectedKeyChange) > 1) { // imperfect intervals
192 expectedKeyChange -= STANDARD_DIATONIC_STEPS;
193 }
194 }
195 int alteration = (keySigChange - expectedKeyChange) / STANDARD_DIATONIC_STEPS;
196 return alteration;
197}

206class Transposer
207{
208private:
209 int m_displacement;
210 int m_alteration; // alteration from key signature
211 int m_numberOfEdoDivisions; // number of divisions in the EDO (default 12)
212 std::vector<int> m_keyMap; // step map for the EDO
213
214public:
223 Transposer(int displacement, int alteration,
224 bool isMinor = false, int numberOfEdoDivisions = STANDARD_12EDO_STEPS,
225 const std::optional <std::vector<int>>& keyMap = std::nullopt)
226 : m_displacement(displacement), m_alteration(alteration), m_numberOfEdoDivisions(numberOfEdoDivisions)
227 {
228 if (keyMap) {
229 if (keyMap.value().size() != STANDARD_DIATONIC_STEPS) {
230 throw std::invalid_argument("The Transposer class only supports key map arrays of " + std::to_string(STANDARD_DIATONIC_STEPS) + " elements");
231 }
232 m_keyMap = keyMap.value();
233 } else if (isMinor) {
234 m_keyMap.assign(MINOR_KEYMAP.begin(), MINOR_KEYMAP.end());
235 } else {
236 m_keyMap.assign(MAJOR_KEYMAP.begin(), MAJOR_KEYMAP.end());
237 }
238 }
239
241 int displacement() const { return m_displacement; }
242
244 int alteration() const { return m_alteration; }

248 void diatonicTranspose(int interval)
249 {
250 m_displacement += interval;
251 }
252
255 void enharmonicTranspose(int direction)
256 {
257 const int keyStepEnharmonic = calcStepsBetweenScaleDegrees(m_displacement, m_displacement + sign(direction));
258 diatonicTranspose(sign(direction));
259 m_alteration -= sign(direction) * keyStepEnharmonic;
260 }

283 void chromaticTranspose(int interval, int chromaticAlteration)
284 {
285 const int intervalNormalized = signedModulus(interval, STANDARD_DIATONIC_STEPS);
286 const int stepsInAlteration = calcStepsInAlteration(interval, chromaticAlteration);
287 const int stepsInInterval = calcStepsInNormalizedInterval(intervalNormalized);
288 const int stepsInDiatonicInterval = calcStepsBetweenScaleDegrees(m_displacement, m_displacement + intervalNormalized);
289
290 const int effectiveAlteration = stepsInAlteration + stepsInInterval - sign(interval) * stepsInDiatonicInterval;
291
292 diatonicTranspose(interval);
293 m_alteration += effectiveAlteration;
294 }

302 void simplifySpelling()
303 {
304 while (std::abs(m_alteration) > 0) {
305 const int currSign = sign(m_alteration);
306 const int currAbsDisp = std::abs(m_alteration);
307 enharmonicTranspose(currSign);
308 if (std::abs(m_alteration) >= currAbsDisp) {
309 enharmonicTranspose(-currSign);
310 return;
311 }
312 if (currSign != sign(m_alteration)) {
313 break;
314 }
315 }
316 }

327 void stepwiseTranspose(int numberOfEdoDivisions)
328 {

343 bool isEnharmonicEquivalent(int displacement, int alteration) const {
344 return calcAbsoluteDivision(displacement, alteration) == calcAbsoluteDivision(m_displacement, m_alteration);
345 }
346
347private:
348 int calcFifthSteps() const
349 {
350 // std::log(3.0 / 2.0) / std::log(2.0) is 0.5849625007211562.
351 static constexpr double kFifthsMultiplier = 0.5849625007211562;
352 return static_cast<int>(std::floor(m_numberOfEdoDivisions * kFifthsMultiplier) + 0.5);
353 }
354
355 int calcScaleDegree(int interval) const
356 { return positiveModulus(interval, int(m_keyMap.size())); }
357
358 int calcStepsBetweenScaleDegrees(int firstDisplacement, int secondDisplacement) const
359 {
360 const int firstScaleDegree = calcScaleDegree(firstDisplacement);
361 const int secondScaleDegree = calcScaleDegree(secondDisplacement);
362 int result = sign(secondDisplacement - firstDisplacement) * (m_keyMap[secondScaleDegree] - m_keyMap[firstScaleDegree]);
363 if (result < 0) {
364 result += m_numberOfEdoDivisions;
365 }
366 return result;
367 }
368
369 int calcStepsInAlteration(int interval, int alteration) const
370 {
371 const int fifthSteps = calcFifthSteps();
372 const int plusFifths = sign(interval) * alteration * 7; // number of fifths to add for a chromatic halfstep alteration (in any EDO)
373 const int minusOctaves = sign(interval) * alteration * -4; // number of octaves to subtract for a chromatic halfstep alteration (in any EDO)
374 const int result = sign(interval) * ((plusFifths * fifthSteps) + (minusOctaves * m_numberOfEdoDivisions));
375 return result;
376 }
377
378 int calcStepsInNormalizedInterval(int intervalNormalized) const
379 {
380 const int fifthSteps = calcFifthSteps();
381 const int index = std::abs(intervalNormalized);
382 const int plusFifths = DIATONIC_INTERVAL_ADJUSTMENTS[index][0]; // number of fifths
383 const int minusOctaves = DIATONIC_INTERVAL_ADJUSTMENTS[index][1]; // number of octaves
384
385 return sign(intervalNormalized) * ((plusFifths * fifthSteps) + (minusOctaves * m_numberOfEdoDivisions));
386 }
387
388 int calcAbsoluteDivision(int displacement, int alteration) const {
389 const int scaleDegree = calcScaleDegree(displacement); // 0..6
390 const int baseStep = m_keyMap[scaleDegree];
391
392 const int octaveCount = (displacement < 0 && displacement % STANDARD_DIATONIC_STEPS != 0)
393 ? (displacement / STANDARD_DIATONIC_STEPS) - 1
394 : (displacement / STANDARD_DIATONIC_STEPS);
395 const int octaveSteps = octaveCount * m_numberOfEdoDivisions;
396 const int chromaticSteps = calcStepsInAlteration(/*interval=*/+1, alteration);
397
398 return baseStep + chromaticSteps + octaveSteps;
399 }
