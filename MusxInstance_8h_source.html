61template <typename T>
62class MusxInstanceListBase : public std::vector<MusxInstance<T>>
63{
64 using VectorType = std::vector<MusxInstance<T>>;
65
66public:
68 explicit MusxInstanceListBase(const std::weak_ptr<Document>& document, Cmper partId)
69 : m_document(document), m_partId(partId) {}
70
72 Cmper getRequestedPartId() const { return m_partId; }
73
75 std::shared_ptr<Document> getDocument() const
76 {
77 auto document = m_document.lock();
78 MUSX_ASSERT_IF(!document) {
79 throw std::logic_error("Document pointer is no longer valid.");
80 }
81 return document;
82 }
83
84private:
85 std::weak_ptr<Document> m_document;
86 Cmper m_partId;
87};
