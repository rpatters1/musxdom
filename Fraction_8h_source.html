38class Fraction {
39private:
40 int m_numerator;
41 int m_denominator;
42
47 void reduce() {
48 int gcd = std::gcd(m_numerator, m_denominator);
49 m_numerator /= gcd;
50 m_denominator /= gcd;
51
52 // Ensure the m_denominator is always positive
53 if (m_denominator < 0) {
54 m_numerator = -m_numerator;
55 m_denominator = -m_denominator;
56 }
57 }
58
59public:
66 Fraction(int num = 0, int den = 1) : m_numerator(num), m_denominator(den) {
67 if (m_denominator == 0) {
68 throw std::invalid_argument("Denominator cannot be zero.");
69 }
70 reduce();
71 }
