38class [[nodiscard]] Fraction {
39private:
40 int m_numerator = 0;
41 int m_denominator = 1;
42
47 static constexpr std::pair<int, int> reduce(int num, int den) {
48 int gcd = std::gcd(num, den);
49 num /= gcd;
50 den /= gcd;
51
52 // Ensure denominator is always positive
53 if (den < 0) {
54 num = -num;
55 den = -den;
56 }
57
58 return {num, den};
59 }
60
61 constexpr static Fraction fromConstExpr(int num, int den)
62 {
63 auto [n, d] = reduce(num, den);
64 auto result = Fraction(n);
65 result.m_denominator = d;
66 return result;
67 }
68
69public:
70 constexpr Fraction() = default;
71
79 Fraction(int num, int den)
80 {
81 if (den == 0) {
82 throw std::invalid_argument("Denominator cannot be zero.");
83 }
84
85 auto [n, d] = reduce(num, den);
86 m_numerator = n;
87 m_denominator = d;
88 }
89
95 constexpr Fraction(int value) : m_numerator(value), m_denominator(1) {}
96
101 static constexpr Fraction fromEdu(dom::Edu edu) { return fromConstExpr(edu, EDU_PER_WHOLE_NOTE); }
102
104 static constexpr Fraction max() noexcept
105 {
106 return Fraction(std::numeric_limits<int>::max());
107 }
108
111 static constexpr Fraction fromPercent(int percent) { return fromConstExpr(percent, 100); }
112
117 constexpr int numerator() const { return m_numerator; }
118
123 constexpr int denominator() const { return m_denominator; }
124
129 constexpr int quotient() const {
130 return m_numerator / m_denominator;
131 }
132
137 Fraction constexpr remainder() const {
138 return fromConstExpr(m_numerator % m_denominator, m_denominator);
139 }
140
143 Fraction reciprocal() const {
144 return Fraction(m_denominator, m_numerator);
145 }
146
150 constexpr dom::Edu calcEduDuration() const {
151 const int num = numerator() * EDU_PER_WHOLE_NOTE;
152 const int den = denominator();
153 const double div = double(num) / double(den);
154 return dom::Edu(static_cast<int>(div + (div >= 0.0 ? 0.5 : -0.5)));
155 }
156
160 constexpr Fraction abs() const
161 {
162 if (numerator() < 0)
163 return fromConstExpr(-numerator(), denominator());
164 return *this;
165 }
166
170 constexpr double toDouble() const {
171 return double(m_numerator) / double(m_denominator);
172 }
173
179 Fraction constexpr operator+(const Fraction& other) const {
180 return fromConstExpr(
181 m_numerator * other.m_denominator + other.m_numerator * m_denominator,
182 m_denominator * other.m_denominator
183 );
184 }
185
191 Fraction constexpr operator-(const Fraction& other) const {
192 return fromConstExpr(
193 m_numerator * other.m_denominator - other.m_numerator * m_denominator,
194 m_denominator * other.m_denominator
195 );
196 }
197
203 Fraction constexpr operator*(const Fraction& other) const {
204 return fromConstExpr(
205 m_numerator * other.m_numerator,
206 m_denominator * other.m_denominator
207 );
208 }
209
217 Fraction operator/(const Fraction& other) const {
218 return Fraction(
219 m_numerator * other.m_denominator,
220 m_denominator * other.m_numerator
221 );
222 }

271 return m_numerator == other.m_numerator && m_denominator == other.m_denominator;
272 }

279 constexpr bool operator!=(const Fraction& other) const {
280 return !(*this == other);

288 constexpr bool operator<(const Fraction& other) const {
289 double lhs = static_cast<double>(m_numerator) / m_denominator;
290 double rhs = static_cast<double>(other.m_numerator) / other.m_denominator;
291 return lhs < rhs;
292 }
293
299 constexpr bool operator<=(const Fraction& other) const {
300 return *this < other || *this == other;
301 }

308 constexpr bool operator>(const Fraction& other) const {
309 return !(*this <= other);

318 return !(*this < other);

325 constexpr explicit operator bool() const {
326 return m_numerator != 0;

335 friend std::ostream& operator<<(std::ostream& os, const Fraction& frac) {
336 os << frac.m_numerator;
337 if (frac.m_denominator != 1) {
338 os << "/" << frac.m_denominator;
339 }
340 return os;
341 }
342
350 friend std::istream& operator>>(std::istream& is, Fraction& frac) {
351 int num, den;
352 char sep;
353 is >> num >> sep >> den;
354 if (sep != '/' || den == 0) {
355 throw std::invalid_argument("Invalid fraction format or zero m_denominator.");
356 }
357 frac = Fraction(num, den);
358 return is;
359 }
