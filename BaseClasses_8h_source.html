280template <typename ElementType, size_t REQUIRED_SIZE = 0>
281class OthersArray : public OthersBase
282{
283private:
284 virtual std::string_view xmlTag() const = 0;
285
286public:
288 explicit OthersArray(const DocumentWeakPtr& document, Cmper partId, ShareMode shareMode, Cmper cmper)
289 : OthersBase(document, partId, shareMode, cmper)
290 {
291 }
292
293 std::vector<ElementType> values;
295
297 void integrityCheck() override
298 {
299 OthersBase::integrityCheck();
300 if constexpr (REQUIRED_SIZE) {
301 const size_t originalSize = values.size();
302 values.resize(REQUIRED_SIZE); // resize first, in case MUSX_INTEGRITY_ERROR throws. (Avoid unreachable code warning.)
303 if (originalSize < REQUIRED_SIZE) {
304 MUSX_INTEGRITY_ERROR("Array with xml tag " + std::string(xmlTag()) + " and cmper " + std::to_string(getCmper())
305 + " has fewer than " + std::to_string(REQUIRED_SIZE) + " elements.");
306 }
307 }
308 }
