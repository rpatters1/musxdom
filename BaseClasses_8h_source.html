236
238 template <typename ParentClass = Base>
239 MusxInstance<ParentClass> getParent() const
240 {
241 auto result = m_parent.lock();
242 MUSX_ASSERT_IF (!result) {
243 throw std::logic_error("Attempt to get parent of contained class, but the parent is no longer allocated.");
244 }
245 if constexpr (std::is_same_v<Base, ParentClass>) {
246 return result;
247 } else {
248 return std::dynamic_pointer_cast<const ParentClass>(result);
249 }
250 }
251
252private:
253 const MusxInstanceWeak<Base> m_parent;
254};

370 {

372 return std::find(v.begin(), v.end(), value) != v.end();
373 }
374
376 void integrityCheck(const std::shared_ptr<Base>& ptrToThis) override
377 {
378 OthersBase::integrityCheck(ptrToThis);
379 if constexpr (REQUIRED_SIZE > 0) {
380 const size_t originalSize = values.size();
381 values.resize(REQUIRED_SIZE); // resize first, in case MUSX_INTEGRITY_ERROR throws. (Avoid unreachable code warning.)
382 if (originalSize < REQUIRED_SIZE) {
383 MUSX_INTEGRITY_ERROR("Array with xml tag " + std::string(xmlTag()) + " and cmper " + std::to_string(getCmper())
384 + " has fewer than " + std::to_string(REQUIRED_SIZE) + " elements.");
