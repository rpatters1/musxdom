205class ContainedClassBase : public Base
206{
207public:
213 ContainedClassBase(const std::shared_ptr<Base>& parent)
214 : Base(parent->getDocument(), SCORE_PARTID, ShareMode::All), m_parent(parent)
215 {}
216
218 Cmper getPartId() const override { return getParent()->getPartId(); }
219
221 template <typename ParentClass = Base>
222 std::shared_ptr<ParentClass> getParent() const
223 {
224 auto result = m_parent.lock();
225 MUSX_ASSERT_IF (!result) {
226 throw std::logic_error("Attempt to get parent of contained class, but the parent is no longer allocated.");
227 }
228 if constexpr (std::is_same_v<Base, ParentClass>) {
229 return result;
230 } else {
231 return std::dynamic_pointer_cast<ParentClass>(result);
232 }
233 }
234
235private:
236 const std::weak_ptr<Base> m_parent;
237};

303{

306
307public:
309 explicit OthersArray(const DocumentWeakPtr& document, Cmper partId, ShareMode shareMode, Cmper cmper)
310 : OthersBase(document, partId, shareMode, cmper)
311 {
312 }
313
314 std::vector<ElementType> values;
316
318 void integrityCheck() override
319 {
320 OthersBase::integrityCheck();
321 if constexpr (REQUIRED_SIZE > 0) {
322 const size_t originalSize = values.size();
323 values.resize(REQUIRED_SIZE); // resize first, in case MUSX_INTEGRITY_ERROR throws. (Avoid unreachable code warning.)
324 if (originalSize < REQUIRED_SIZE) {
325 MUSX_INTEGRITY_ERROR("Array with xml tag " + std::string(xmlTag()) + " and cmper " + std::to_string(getCmper())
326 + " has fewer than " + std::to_string(REQUIRED_SIZE) + " elements.");
327 }
328 }
