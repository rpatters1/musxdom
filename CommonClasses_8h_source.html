155class KeySignature : public Base
156{
157private:
158 std::vector<unsigned> calcTonalCenterArray() const;
159 std::vector<int> calcAcciAmountsArray() const;
160 std::vector<unsigned> calcAcciOrderArray() const;
161
162public:
167 explicit KeySignature(const DocumentWeakPtr& document)
168 : Base(document, SCORE_PARTID, ShareMode::All) {}
169
176 uint16_t key{};
177 bool keyless{};
178 bool hideKeySigShowAccis{};
179
194 Cmper getKeyMode() const { return isLinear() ? key >> 8 : key; }
195
198 std::optional<int> getAlteration() const { return isLinear() ? std::optional<int>(int8_t(key & 0xff)) : std::nullopt; }
199
200 bool isLinear() const { return (key & 0xC000) == 0; }
201 bool isNonLinear() const { return (key & 0xC000) != 0; }
202 bool isBuiltIn() const { return isLinear() && getKeyMode() <= 1; }
203 bool isMajor() const { return getKeyMode() == 0; }
204 bool isMinor() const { return getKeyMode() == 1; }
205
207 bool isSame(const KeySignature& src)
208 {
209 return key == src.key && keyless == src.keyless && hideKeySigShowAccis == src.hideKeySigShowAccis;
210 }
211
217 int calcTonalCenterIndex() const;
218
221 int calcAlterationOnNote(unsigned noteIndex) const;
222
223 void integrityCheck() override
224 {
225 Base::integrityCheck();
226 if (key >= 0x8000) {
227 MUSX_INTEGRITY_ERROR("Key signature has invalid key value: " + std::to_string(key));
228 }
229 }
230
231 static const xml::XmlElementArray<KeySignature>& xmlMappingArray();
232};

242class TimeSignature : public Base
243{
244public:
245
247 struct TimeSigComponent
248 {
249 std::vector<util::Fraction> counts;
251 std::vector<Edu> units;
254
256 bool operator==(const TimeSigComponent& src) const
257 { return counts == src.counts && units == src.units; }
258
260 util::Fraction sumCounts() const
261 { return std::accumulate(counts.begin(), counts.end(), util::Fraction{}); }
262
264 Edu sumUnits() const
265 { return std::accumulate(units.begin(), units.end(), Edu{}); }
266 };
267
268 std::vector<TimeSigComponent> components;
269
274 std::pair<util::Fraction, NoteType> calcSimplified() const;
275
277 util::Fraction calcTotalDuration() const
278 {
279 util::Fraction result = std::accumulate(components.begin(), components.end(), util::Fraction{},
280 [](const util::Fraction& acc, const TimeSigComponent& comp)
281 { return acc + (comp.sumCounts() * comp.sumUnits()); }
282 );
283 return result / Edu(NoteType::Whole);
284 }
285
287 bool isSame(const TimeSignature& src)
288 {
289 return components == src.components && m_abbreviate == src.m_abbreviate;
290 }
291
295 std::shared_ptr<TimeSignature> createComponent(size_t index) const
296 {
297 checkIndex(index);
298 return std::shared_ptr<TimeSignature>(new TimeSignature(getDocument(), components[index], m_abbreviate));
299 }
300
305 std::optional<char32_t> getAbbreviatedSymbol() const;
306
308 bool isCommonTime() const;
310 bool isCutTime() const;
311
312private:
313 void checkIndex(size_t index) const
314 {
315 if (index > components.size()) {
316 throw std::invalid_argument("Index out of range. The time signature has " + std::to_string(components.size())
317 + " elements. The index requested was " + std::to_string(index) + ".");
318 }
319 }
320
325 explicit TimeSignature(const DocumentWeakPtr& document, int beats, Edu unit, bool hasCompositeTop, bool hasCompositeBottom,
326 std::optional<bool> abbreviate = std::nullopt);
327
332 explicit TimeSignature(const DocumentWeakPtr& document, const TimeSigComponent& timeSigUnit, std::optional<bool> abbreviate)
333 : Base(document, SCORE_PARTID, ShareMode::All), m_abbreviate(abbreviate)
334 {
335 components.push_back(timeSigUnit);
336 }
337
338 std::optional<bool> m_abbreviate;
339
340 friend class others::Measure;
341};
