340class TimeSignature : public CommonClassBase
341{
342public:
343
345 struct TimeSigComponent
346 {
347 std::vector<util::Fraction> counts;
349 std::vector<Edu> units;
352
354 bool operator==(const TimeSigComponent& src) const
355 { return counts == src.counts && units == src.units; }
356
358 util::Fraction sumCounts() const
359 { return std::accumulate(counts.begin(), counts.end(), util::Fraction{}); }
360
362 Edu sumUnits() const
363 { return std::accumulate(units.begin(), units.end(), Edu{}); }
364 };
365
366 std::vector<TimeSigComponent> components;
367
372 std::pair<util::Fraction, NoteType> calcSimplified() const;
373
375 util::Fraction calcTotalDuration() const
376 {
377 util::Fraction result = std::accumulate(components.begin(), components.end(), util::Fraction{},
378 [](const util::Fraction& acc, const TimeSigComponent& comp)
379 { return acc + (comp.sumCounts() * comp.sumUnits()); }
380 );
381 return result / Edu(NoteType::Whole);
382 }
383
385 bool isSame(const TimeSignature& src)
386 {
387 return components == src.components && m_abbreviate == src.m_abbreviate;
388 }
389
393 std::shared_ptr<TimeSignature> createComponent(size_t index) const
394 {
395 checkIndex(index);
396 return std::shared_ptr<TimeSignature>(new TimeSignature(getDocument(), components[index], m_abbreviate));
397 }
398
403 std::optional<char32_t> getAbbreviatedSymbol() const;
404
406 bool isCommonTime() const;
408 bool isCutTime() const;
409
410private:
411 void checkIndex(size_t index) const
412 {
413 if (index > components.size()) {
414 throw std::invalid_argument("Index out of range. The time signature has " + std::to_string(components.size())
415 + " elements. The index requested was " + std::to_string(index) + ".");
416 }
417 }
418
423 explicit TimeSignature(const DocumentWeakPtr& document, int beats, Edu unit, bool hasCompositeTop, bool hasCompositeBottom,
424 std::optional<bool> abbreviate = std::nullopt);
425
430 explicit TimeSignature(const DocumentWeakPtr& document, const TimeSigComponent& timeSigUnit, std::optional<bool> abbreviate)
431 : CommonClassBase(document), m_abbreviate(abbreviate)
432 {
433 components.push_back(timeSigUnit);
434 }
435
436 std::optional<bool> m_abbreviate;
437
438 friend class others::Measure;
439 friend class details::IndependentStaffDetails;
440};
