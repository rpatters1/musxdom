528class TimeSignature : public CommonClassBase
529{
530public:
531
533 struct TimeSigComponent
534 {
535 std::vector<util::Fraction> counts;
537 std::vector<Edu> units;
540
542 bool operator==(const TimeSigComponent& src) const
543 { return counts == src.counts && units == src.units; }
544
546 util::Fraction sumCounts() const
547 { return std::accumulate(counts.begin(), counts.end(), util::Fraction{}); }
548
550 Edu sumUnits() const
551 { return std::accumulate(units.begin(), units.end(), Edu{}); }
552 };
553
554 std::vector<TimeSigComponent> components;
555
560 std::pair<util::Fraction, NoteType> calcSimplified() const;
561
563 util::Fraction calcTotalDuration() const
564 {
565 util::Fraction result = std::accumulate(components.begin(), components.end(), util::Fraction{},
566 [](const util::Fraction& acc, const TimeSigComponent& comp)
567 { return acc + (comp.sumCounts() * comp.sumUnits()); }
568 );
569 return result / Edu(NoteType::Whole);
570 }
571
573 bool isSame(const TimeSignature& src) const;
574
578 MusxInstance<TimeSignature> createComponent(size_t index) const
579 {
580 checkIndex(index);
581 return MusxInstance<TimeSignature>(new TimeSignature(getDocument(), components[index], m_abbreviate));
582 }
583
588 std::optional<char32_t> getAbbreviatedSymbol() const;
589
591 bool isCommonTime() const;
593 bool isCutTime() const;
594
595private:
596 void checkIndex(size_t index) const
597 {
598 if (index > components.size()) {
599 throw std::invalid_argument("Index out of range. The time signature has " + std::to_string(components.size())
600 + " elements. The index requested was " + std::to_string(index) + ".");
601 }
602 }
603
608 explicit TimeSignature(const DocumentWeakPtr& document, int beats, Edu unit, bool hasCompositeTop, bool hasCompositeBottom,
609 std::optional<bool> abbreviate = std::nullopt);
610
615 explicit TimeSignature(const DocumentWeakPtr& document, const TimeSigComponent& timeSigUnit, std::optional<bool> abbreviate)
616 : CommonClassBase(document), m_abbreviate(abbreviate)
617 {
618 components.push_back(timeSigUnit);
619 }
620
621 std::optional<bool> m_abbreviate;
622
623 friend class others::Measure;
624 friend class others::OssiaHeader;
625 friend class details::IndependentStaffDetails;
626};
