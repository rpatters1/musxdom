413class TimeSignature : public CommonClassBase
414{
415public:
416
418 struct TimeSigComponent
419 {
420 std::vector<util::Fraction> counts;
422 std::vector<Edu> units;
425
427 bool operator==(const TimeSigComponent& src) const
428 { return counts == src.counts && units == src.units; }
429
431 util::Fraction sumCounts() const
432 { return std::accumulate(counts.begin(), counts.end(), util::Fraction{}); }
433
435 Edu sumUnits() const
436 { return std::accumulate(units.begin(), units.end(), Edu{}); }
437 };
438
439 std::vector<TimeSigComponent> components;
440
445 std::pair<util::Fraction, NoteType> calcSimplified() const;
446
448 util::Fraction calcTotalDuration() const
449 {
450 util::Fraction result = std::accumulate(components.begin(), components.end(), util::Fraction{},
451 [](const util::Fraction& acc, const TimeSigComponent& comp)
452 { return acc + (comp.sumCounts() * comp.sumUnits()); }
453 );
454 return result / Edu(NoteType::Whole);
455 }
456
458 bool isSame(const TimeSignature& src)
459 {
460 return components == src.components && m_abbreviate == src.m_abbreviate;
461 }
462
466 std::shared_ptr<TimeSignature> createComponent(size_t index) const
467 {
468 checkIndex(index);
469 return std::shared_ptr<TimeSignature>(new TimeSignature(getDocument(), components[index], m_abbreviate));
470 }
471
476 std::optional<char32_t> getAbbreviatedSymbol() const;
477
479 bool isCommonTime() const;
481 bool isCutTime() const;
482
483private:
484 void checkIndex(size_t index) const
485 {
486 if (index > components.size()) {
487 throw std::invalid_argument("Index out of range. The time signature has " + std::to_string(components.size())
488 + " elements. The index requested was " + std::to_string(index) + ".");
489 }
490 }
491
496 explicit TimeSignature(const DocumentWeakPtr& document, int beats, Edu unit, bool hasCompositeTop, bool hasCompositeBottom,
497 std::optional<bool> abbreviate = std::nullopt);
498
503 explicit TimeSignature(const DocumentWeakPtr& document, const TimeSigComponent& timeSigUnit, std::optional<bool> abbreviate)
504 : CommonClassBase(document), m_abbreviate(abbreviate)
505 {
506 components.push_back(timeSigUnit);
507 }
508
509 std::optional<bool> m_abbreviate;
510
511 friend class others::Measure;
512 friend class details::IndependentStaffDetails;
513};
