465class TimeSignature : public CommonClassBase
466{
467public:
468
470 struct TimeSigComponent
471 {
472 std::vector<util::Fraction> counts;
474 std::vector<Edu> units;
477
479 bool operator==(const TimeSigComponent& src) const
480 { return counts == src.counts && units == src.units; }
481
483 util::Fraction sumCounts() const
484 { return std::accumulate(counts.begin(), counts.end(), util::Fraction{}); }
485
487 Edu sumUnits() const
488 { return std::accumulate(units.begin(), units.end(), Edu{}); }
489 };
490
491 std::vector<TimeSigComponent> components;
492
497 std::pair<util::Fraction, NoteType> calcSimplified() const;
498
500 util::Fraction calcTotalDuration() const
501 {
502 util::Fraction result = std::accumulate(components.begin(), components.end(), util::Fraction{},
503 [](const util::Fraction& acc, const TimeSigComponent& comp)
504 { return acc + (comp.sumCounts() * comp.sumUnits()); }
505 );
506 return result / Edu(NoteType::Whole);
507 }
508
510 bool isSame(const TimeSignature& src) const;
511
515 MusxInstance<TimeSignature> createComponent(size_t index) const
516 {
517 checkIndex(index);
518 return MusxInstance<TimeSignature>(new TimeSignature(getDocument(), components[index], m_abbreviate));
519 }
520
525 std::optional<char32_t> getAbbreviatedSymbol() const;
526
528 bool isCommonTime() const;
530 bool isCutTime() const;
531
532private:
533 void checkIndex(size_t index) const
534 {
535 if (index > components.size()) {
536 throw std::invalid_argument("Index out of range. The time signature has " + std::to_string(components.size())
537 + " elements. The index requested was " + std::to_string(index) + ".");
538 }
539 }
540
545 explicit TimeSignature(const DocumentWeakPtr& document, int beats, Edu unit, bool hasCompositeTop, bool hasCompositeBottom,
546 std::optional<bool> abbreviate = std::nullopt);
547
552 explicit TimeSignature(const DocumentWeakPtr& document, const TimeSigComponent& timeSigUnit, std::optional<bool> abbreviate)
553 : CommonClassBase(document), m_abbreviate(abbreviate)
554 {
555 components.push_back(timeSigUnit);
556 }
557
558 std::optional<bool> m_abbreviate;
559
560 friend class others::Measure;
561 friend class others::OssiaHeader;
562 friend class details::IndependentStaffDetails;
563};
