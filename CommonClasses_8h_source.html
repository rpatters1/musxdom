556class TimeSignature : public DocumentElementNoPart
557{
558public:
561 enum class Abbreviation
562 {
563 NotApplicable,
564 Numeric,
565 Abbreviated
566 };
567
569 struct TimeSigComponent
570 {
571 std::vector<util::Fraction> counts;
573 std::vector<Edu> units;
576
578 bool operator==(const TimeSigComponent& src) const
579 { return counts == src.counts && units == src.units; }
580
582 util::Fraction sumCounts() const
583 { return std::accumulate(counts.begin(), counts.end(), util::Fraction{}); }
584
586 Edu sumUnits() const
587 { return std::accumulate(units.begin(), units.end(), Edu{}); }
588 };
589
590 std::vector<TimeSigComponent> components;
591
596 std::pair<util::Fraction, NoteType> calcSimplified() const;
597
599 util::Fraction calcTotalDuration() const
600 {
601 util::Fraction result = std::accumulate(components.begin(), components.end(), util::Fraction{},
602 [](const util::Fraction& acc, const TimeSigComponent& comp)
603 { return acc + (comp.sumCounts() * comp.sumUnits()); }
604 );
605 return result / Edu(NoteType::Whole);
606 }
607
617 util::Fraction calcBeatValueAt(Edu eduPosition) const;
618
620 bool isSame(const TimeSignature& src) const;
621
625 MusxInstance<TimeSignature> createComponent(size_t index) const
626 {
627 checkIndex(index);
628 return MusxInstance<TimeSignature>(new TimeSignature(getDocument(), components[index], m_abbreviation));
629 }
630
635 std::optional<char32_t> getAbbreviatedSymbol() const;
636
638 bool isCommonTime() const;
640 bool isCutTime() const;
641
642private:
643 void checkIndex(size_t index) const
644 {
645 if (index > components.size()) {
646 throw std::invalid_argument("Index out of range. The time signature has " + std::to_string(components.size())
647 + " elements. The index requested was " + std::to_string(index) + ".");
648 }
649 }
650
652 explicit TimeSignature(const DocumentWeakPtr& document, int beats, Edu unit, bool hasCompositeTop, bool hasCompositeBottom,
653 Abbreviation abbreviate = {});
654
656 explicit TimeSignature(const DocumentWeakPtr& document, const TimeSigComponent& timeSigUnit, Abbreviation abbreviate = {})
657 : DocumentElementNoPart(document), m_abbreviation(abbreviate)
658 {
659 components.push_back(timeSigUnit);
660 }
661
662 Abbreviation m_abbreviation{};
663
664 friend class others::Measure;
665 friend class others::OssiaHeader;
666 friend class details::IndependentStaffDetails;
667};
