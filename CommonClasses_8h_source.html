169
184 Cmper getKeyMode() const { return isLinear() ? key >> 8 : key; }
185
188 std::optional<int> getAlteration() const { return isLinear() ? std::optional<int>(int8_t(key & 0xff)) : std::nullopt; }
189
190 bool isLinear() const { return (key & 0xC000) == 0; }
191 bool isNonLinear() const { return (key & 0xC000) != 0; }
192 bool isBuiltIn() const { return isLinear() && getKeyMode() <= 1; }
193 bool isMajor() const { return getKeyMode() == 0; }
194 bool isMinor() const { return getKeyMode() == 1; }
195
197 bool isSame(const KeySignature& src)
198 {
199 return key == src.key && keyless == src.keyless && hideKeySigShowAccis == src.hideKeySigShowAccis;
200 }
201
207 int calcTonalCenterIndex() const;
208
211 int calcAlterationOnNote(unsigned noteIndex) const;
212
213 void integrityCheck() override
214 {
215 Base::integrityCheck();
216 if (key >= 0x8000) {
217 MUSX_INTEGRITY_ERROR("Key signature has invalid key value: " + std::to_string(key));
218 }
219 }

262class TimeSignature : public CommonClassBase
263{
264public:
265
267 struct TimeSigComponent
268 {
269 std::vector<util::Fraction> counts;
271 std::vector<Edu> units;
274
276 bool operator==(const TimeSigComponent& src) const
277 { return counts == src.counts && units == src.units; }
278
280 util::Fraction sumCounts() const
281 { return std::accumulate(counts.begin(), counts.end(), util::Fraction{}); }
282
284 Edu sumUnits() const
285 { return std::accumulate(units.begin(), units.end(), Edu{}); }
286 };
287
288 std::vector<TimeSigComponent> components;
289
294 std::pair<util::Fraction, NoteType> calcSimplified() const;
295
297 util::Fraction calcTotalDuration() const
298 {
299 util::Fraction result = std::accumulate(components.begin(), components.end(), util::Fraction{},
300 [](const util::Fraction& acc, const TimeSigComponent& comp)
301 { return acc + (comp.sumCounts() * comp.sumUnits()); }
302 );
303 return result / Edu(NoteType::Whole);
304 }
305
307 bool isSame(const TimeSignature& src)
308 {
309 return components == src.components && m_abbreviate == src.m_abbreviate;
310 }
311
315 std::shared_ptr<TimeSignature> createComponent(size_t index) const
316 {
317 checkIndex(index);
318 return std::shared_ptr<TimeSignature>(new TimeSignature(getDocument(), components[index], m_abbreviate));
319 }
320
325 std::optional<char32_t> getAbbreviatedSymbol() const;
326
328 bool isCommonTime() const;
330 bool isCutTime() const;
331
332private:
333 void checkIndex(size_t index) const
334 {
335 if (index > components.size()) {
336 throw std::invalid_argument("Index out of range. The time signature has " + std::to_string(components.size())
337 + " elements. The index requested was " + std::to_string(index) + ".");
338 }
339 }
340
345 explicit TimeSignature(const DocumentWeakPtr& document, int beats, Edu unit, bool hasCompositeTop, bool hasCompositeBottom,
346 std::optional<bool> abbreviate = std::nullopt);
347
352 explicit TimeSignature(const DocumentWeakPtr& document, const TimeSigComponent& timeSigUnit, std::optional<bool> abbreviate)
353 : CommonClassBase(document), m_abbreviate(abbreviate)
354 {
355 components.push_back(timeSigUnit);
356 }
357
358 std::optional<bool> m_abbreviate;
359
360 friend class others::Measure;
361};
