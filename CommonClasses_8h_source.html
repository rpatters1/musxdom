334class TimeSignature : public CommonClassBase
335{
336public:
337
339 struct TimeSigComponent
340 {
341 std::vector<util::Fraction> counts;
343 std::vector<Edu> units;
346
348 bool operator==(const TimeSigComponent& src) const
349 { return counts == src.counts && units == src.units; }
350
352 util::Fraction sumCounts() const
353 { return std::accumulate(counts.begin(), counts.end(), util::Fraction{}); }
354
356 Edu sumUnits() const
357 { return std::accumulate(units.begin(), units.end(), Edu{}); }
358 };
359
360 std::vector<TimeSigComponent> components;
361
366 std::pair<util::Fraction, NoteType> calcSimplified() const;
367
369 util::Fraction calcTotalDuration() const
370 {
371 util::Fraction result = std::accumulate(components.begin(), components.end(), util::Fraction{},
372 [](const util::Fraction& acc, const TimeSigComponent& comp)
373 { return acc + (comp.sumCounts() * comp.sumUnits()); }
374 );
375 return result / Edu(NoteType::Whole);
376 }
377
379 bool isSame(const TimeSignature& src)
380 {
381 return components == src.components && m_abbreviate == src.m_abbreviate;
382 }
383
387 std::shared_ptr<TimeSignature> createComponent(size_t index) const
388 {
389 checkIndex(index);
390 return std::shared_ptr<TimeSignature>(new TimeSignature(getDocument(), components[index], m_abbreviate));
391 }
392
397 std::optional<char32_t> getAbbreviatedSymbol() const;
398
400 bool isCommonTime() const;
402 bool isCutTime() const;
403
404private:
405 void checkIndex(size_t index) const
406 {
407 if (index > components.size()) {
408 throw std::invalid_argument("Index out of range. The time signature has " + std::to_string(components.size())
409 + " elements. The index requested was " + std::to_string(index) + ".");
410 }
411 }
412
417 explicit TimeSignature(const DocumentWeakPtr& document, int beats, Edu unit, bool hasCompositeTop, bool hasCompositeBottom,
418 std::optional<bool> abbreviate = std::nullopt);
419
424 explicit TimeSignature(const DocumentWeakPtr& document, const TimeSigComponent& timeSigUnit, std::optional<bool> abbreviate)
425 : CommonClassBase(document), m_abbreviate(abbreviate)
426 {
427 components.push_back(timeSigUnit);
428 }
429
430 std::optional<bool> m_abbreviate;
431
432 friend class others::Measure;
433 friend class details::IndependentStaffDetails;
434};
