1176 MUSX_ASSERT_IF(!mask || mask >= unsigned(NoteType::Eighth)) {
1177 return 0; // invalid mask values have already been flagged and supposedly corrected by #integrityCheck.
1178 }
1179 for (unsigned shift = 0; true; shift++) {

1196 mask = unsigned(NoteType::Eighth) - 1;
