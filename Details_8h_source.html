657 BeamNumber calcLowestBreak() const
658 {
659 MUSX_ASSERT_IF(!mask || mask >= unsigned(NoteType::Eighth)) {
660 return 0; // invalid mask values have already been flagged and supposedly corrected by #integrityCheck.
661 }
662 for (unsigned shift = 0; true; shift++) {
663 if (mask & (unsigned(NoteType::Note16th) >> shift)) {
664 return shift + 2; // the 2nd beam is the 16th beam and the first one we checked.
665 }
666 }
667 assert(false); // should not be able to get here
668 return 0;
669 }

672 {

677 }
678 if (mask >= unsigned(NoteType::Eighth)) {
679 mask = unsigned(NoteType::Eighth) - 1;
680 MUSX_INTEGRITY_ERROR("Secondary beam break for entry" + std::to_string(getEntryNumber()) + " specifies a value that cannot be a secondary beam.");
