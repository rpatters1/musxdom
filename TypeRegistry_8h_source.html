76 static inline const auto registry = []() {
77 return std::unordered_map<std::string_view, VariantType>{
78 {Types::XmlNodeName, VariantType(static_cast<Types*>(nullptr))}...
79 };
80 }();

90 static std::optional<VariantType> findRegisteredType(std::string_view nodeName)
91 {
92 const auto it = registry.find(nodeName);
93 if (it == registry.end()) {
94 return std::nullopt;
95 }
96 return it->second;
97 }

100 template <typename T, typename PoolPtr, typename... Args>
101 static auto getScoreValue(const PoolPtr& pool, Args&&... args)
102 {
103 if constexpr (std::is_same_v<PoolPtr, ::musx::dom::OthersPoolPtr> || std::is_same_v<PoolPtr, ::musx::dom::DetailsPoolPtr>) {
104 return pool->template get<T>(SCORE_PARTID, std::forward<Args>(args)...);
105 } else {
106 return pool->template get<T>(std::forward<Args>(args)...);
107 }
108 }

147 template <typename PoolPtr, typename... Args>
148 static std::optional<CreatedInstanceInfo> createInstance([[maybe_unused]]const PoolPtr& pool, const XmlElementPtr& node, ElementLinker& elementLinker, const DocumentPtr& document, Args&&... args)
149 {
150 auto typePtr = TypeRegistry::findRegisteredType(node->getTagName());
151 if (!typePtr.has_value()) {
152 return std::nullopt; // Type not yet implemented
153 }
154
155 return std::visit(
156 [&](auto const& ptr) -> CreatedInstanceInfo {
157 using T = std::remove_pointer_t<std::remove_reference_t<decltype(ptr)>>;
158 // Only enable this part if T is constructible with Args...
159 if constexpr (std::is_constructible_v<T, const DocumentPtr&, Cmper, Base::ShareMode, Args...>) {
160 auto partAttr = node->findAttribute("part");
161 Cmper partId = partAttr ? partAttr->getValueAs<Cmper>() : SCORE_PARTID; // zero is the score ID
162 auto shareMode = Base::ShareMode::All;
163 if (auto shareAttr = node->findAttribute("shared")) {
164 shareMode = shareAttr->getValueAs<bool>() ? Base::ShareMode::Partial : Base::ShareMode::None;
165 }
166 auto instance = std::make_shared<T>(document, partId, shareMode, std::forward<Args>(args)...);
167 if constexpr (!std::is_same_v<PoolPtr, ::musx::dom::EntryPoolPtr>) {
168 if (instance->getShareMode() == Base::ShareMode::Partial) {
169 for (auto child = node->getFirstChildElement(); child; child = child->getNextSibling()) {
170 instance->addUnlinkedNode(child->getTagName());
171 }
172 auto scoreValue = getScoreValue<T>(pool, std::forward<Args>(args)...);
173 if (scoreValue) {
174 *instance = *scoreValue;
175 } else {
176 throw std::invalid_argument("Score instance not found for partially linked part instance");
177 }
178 }
179 }
180 factory::FieldPopulator<T>::populate(instance, node, elementLinker);
181 return CreatedInstanceInfo(instance,T::XmlNodeName);
182 } else {
183 assert(false); // This path should never actually be taken, but it is required for std::visit.
184 throw std::logic_error("Type for " + node->getTagName() + " is not constructible with given arguments");
185 }
186 },
187 typePtr.value()
188 );
189 }
190};
