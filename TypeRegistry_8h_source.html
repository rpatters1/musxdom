81 static std::optional<VariantType> findRegisteredType(std::string_view nodeName)
82 {
83 const auto it = registry.find(nodeName);
84 if (it == registry.end()) {
85 return std::nullopt;
86 }
87 return it->second;

105 template <typename PoolPtr, typename... Args>
106 static std::shared_ptr<Base> createInstance(const PoolPtr& pool, const XmlElementPtr& node, ElementLinker& elementLinker, const DocumentPtr& document, Args&&... args)
107 {
108 auto typePtr = TypeRegistry::findRegisteredType(node->getTagName());
109 if (!typePtr.has_value()) {
110 return nullptr; // Type not yet implemented
111 }
112
113 return std::visit(
114 [&](auto const& ptr) -> std::shared_ptr<Base> {
115 using T = std::remove_pointer_t<std::remove_reference_t<decltype(ptr)>>;
116 // Only enable this part if T is constructible with Args...
117 if constexpr (std::is_constructible_v<T, const DocumentPtr&, Cmper, Base::ShareMode, Args...>) {
118 auto partAttr = node->findAttribute("part");
119 Cmper partId = partAttr ? partAttr->getValueAs<Cmper>() : SCORE_PARTID; // zero is the score ID
120 auto shareMode = Base::ShareMode::All;
121 if (auto shareAttr = node->findAttribute("shared")) {
122 shareMode = shareAttr->getValueAs<bool>() ? Base::ShareMode::Partial : Base::ShareMode::None;
123 }
124 auto instance = std::make_shared<T>(document, partId, shareMode, std::forward<Args>(args)...);
125 if constexpr (!std::is_same_v<PoolPtr, EntryPoolPtr>) {
126 if (instance->getShareMode() == Base::ShareMode::Partial) {
127 for (auto child = node->getFirstChildElement(); child; child = child->getNextSibling()) {
128 instance->addUnlinkedNode(child->getTagName());
129 }
130 auto scoreValue = [&]() {
131 if constexpr (std::is_same_v<PoolPtr, OthersPoolPtr> || std::is_same_v<PoolPtr, DetailsPoolPtr>) {
132 return pool->template get<T>(SCORE_PARTID, std::forward<Args>(args)...);
133 } else {
134 return pool->template get<T>(std::forward<Args>(args)...);
135 }
136 }();
137 if (scoreValue) {
138 *instance = *scoreValue;
139 }
140 else {
141 throw std::invalid_argument("Score instance not found for partially linked part instance");
142 }
143 }
144 }
145 factory::FieldPopulator<T>::populate(instance, node, elementLinker);
146 return instance;
147 } else {
148 throw std::runtime_error("Type for " + node->getTagName() + " is not constructible with given arguments");
149 }
150 },
151 typePtr.value()
152 );
153 }
154};
