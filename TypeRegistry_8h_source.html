75 static inline const auto registry = []() {
76 return std::unordered_map<std::string_view, VariantType>{
77 {Types::XmlNodeName, VariantType(static_cast<Types*>(nullptr))}...
78 };
79 }();

89 static std::optional<VariantType> findRegisteredType(const std::string_view& nodeName)
90 {
91 const auto it = registry.find(nodeName);
92 if (it == registry.end()) {
93 return std::nullopt;
94 }
95 return it->second;

99 template <typename T, typename PoolPtr, typename... Args>
100 static auto getScoreValue(const PoolPtr& pool, Args&&... args)
101 {
102 if constexpr (std::is_same_v<PoolPtr, ::musx::dom::OthersPoolPtr> || std::is_same_v<PoolPtr, ::musx::dom::DetailsPoolPtr>) {
103 return pool->template get<T>(SCORE_PARTID, std::forward<Args>(args)...);
104 } else {
105 return pool->template get<T>(std::forward<Args>(args)...);
106 }
107 }
108
109public:
112 template <typename T>
113 struct is_registered_type : std::disjunction<std::is_same<T, Types>...> {};
114
117 template <typename T>
118 static constexpr bool is_registered_type_v = is_registered_type<T>::value;
119
134 template <typename PoolPtr, typename... Args>
135 static std::shared_ptr<Base> createInstance([[maybe_unused]]const PoolPtr& pool, const XmlElementPtr& node, ElementLinker& elementLinker, const DocumentPtr& document, Args&&... args)
136 {
137 auto typePtr = TypeRegistry::findRegisteredType(node->getTagName());
138 if (!typePtr.has_value()) {
139 return nullptr; // Type not yet implemented
140 }
141
142 return std::visit(
143 [&](auto const& ptr) -> std::shared_ptr<Base> {
144 using T = std::remove_pointer_t<std::remove_reference_t<decltype(ptr)>>;
145 // Only enable this part if T is constructible with Args...
146 if constexpr (std::is_constructible_v<T, const DocumentPtr&, Cmper, Base::ShareMode, Args...>) {
147 auto partAttr = node->findAttribute("part");
148 Cmper partId = partAttr ? partAttr->getValueAs<Cmper>() : SCORE_PARTID; // zero is the score ID
149 auto shareMode = Base::ShareMode::All;
150 if (auto shareAttr = node->findAttribute("shared")) {
151 shareMode = shareAttr->getValueAs<bool>() ? Base::ShareMode::Partial : Base::ShareMode::None;
152 }
153 auto instance = std::make_shared<T>(document, partId, shareMode, std::forward<Args>(args)...);
154 if constexpr (!std::is_same_v<PoolPtr, ::musx::dom::EntryPoolPtr>) {
155 if (instance->getShareMode() == Base::ShareMode::Partial) {
156 for (auto child = node->getFirstChildElement(); child; child = child->getNextSibling()) {
157 instance->addUnlinkedNode(child->getTagName());
158 }
159 auto scoreValue = getScoreValue<T>(pool, std::forward<Args>(args)...);
160 if (scoreValue) {
161 *instance = *scoreValue;
162 } else {
163 throw std::invalid_argument("Score instance not found for partially linked part instance");
164 }
165 }
166 }
167 factory::FieldPopulator<T>::populate(instance, node, elementLinker);
168 return instance;
169 } else {
170 assert(false); // This path should never actually be taken, but it is required for std::visit.
171 throw std::logic_error("Type for " + node->getTagName() + " is not constructible with given arguments");
172 }
173 },
174 typePtr.value()
175 );
176 }
177};
