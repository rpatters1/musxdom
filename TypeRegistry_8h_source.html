103 template <typename PoolPtr, typename... Args>
104 static std::shared_ptr<Base> createInstance(const PoolPtr& pool, const XmlElementPtr& node, ElementLinker& elementLinker, const DocumentPtr& document, Args&&... args)
105 {
106 auto typePtr = TypeRegistry::findRegisteredType(node->getTagName());
107 if (!typePtr.has_value()) {
108 return nullptr; // Type not yet implemented
109 }
110
111 return std::visit(
112 [&](auto const& ptr) -> std::shared_ptr<Base> {
113 using T = std::remove_pointer_t<std::remove_reference_t<decltype(ptr)>>;
114 // Only enable this part if T is constructible with Args...
115 if constexpr (std::is_constructible_v<T, const DocumentPtr&, Cmper, Base::ShareMode, Args...>) {
116 auto partAttr = node->findAttribute("part");
117 Cmper partId = partAttr ? partAttr->getValueAs<Cmper>() : SCORE_PARTID; // zero is the score ID
118 auto shareMode = Base::ShareMode::All;
119 if (auto shareAttr = node->findAttribute("shared")) {
120 shareMode = shareAttr->getValueAs<bool>() ? Base::ShareMode::Partial : Base::ShareMode::None;
121 }
122 auto instance = std::make_shared<T>(document, partId, shareMode, std::forward<Args>(args)...);
123 if constexpr (!std::is_same_v<PoolPtr, EntryPoolPtr>) {
124 if (instance->getShareMode() == Base::ShareMode::Partial) {
125 for (auto child = node->getFirstChildElement(); child; child = child->getNextSibling()) {
126 instance->addUnlinkedNode(child->getTagName());
127 }
128 auto scoreValue = [&]() {
129 if constexpr (std::is_same_v<PoolPtr, OthersPoolPtr> || std::is_same_v<PoolPtr, DetailsPoolPtr>) {
130 return pool->template get<T>(SCORE_PARTID, std::forward<Args>(args)...);
131 } else {
132 return pool->template get<T>(std::forward<Args>(args)...);
133 }
134 }();
135 if (scoreValue) {
136 *instance = *scoreValue;
137 }
138 else {
139 throw std::invalid_argument("Score instance not found for partially linked part instance");
140 }
141 }
142 }
143 factory::FieldPopulator<T>::populate(instance, node, elementLinker);
144 return instance;
145 } else {
146 throw std::runtime_error("Type for " + node->getTagName() + " is not constructible with given arguments");
147 }
148 },
149 typePtr.value()
150 );
151 }
152};
