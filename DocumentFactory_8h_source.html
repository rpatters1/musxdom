51 template <typename XmlDocumentType>
52 static DocumentPtr create(const char * data, size_t size)
53 {
54 static_assert(std::is_base_of<musx::xml::IXmlDocument, XmlDocumentType>::value,
55 "XmlReaderType must derive from IXmlDocument.");
56
57 std::unique_ptr<musx::xml::IXmlDocument> xmlDocument = std::make_unique<XmlDocumentType>();
58 xmlDocument->loadFromBuffer(data, size);
59
60 auto rootElement = xmlDocument->getRootElement();
61 if (!rootElement || rootElement->getTagName() != "finale") {
62 throw std::invalid_argument("Missing <finale> element.");
63 }
64
65 DocumentPtr document(new Document);
66
67 ElementLinker elementLinker;
68 for (auto element = rootElement->getFirstChildElement(); element; element = element->getNextSibling()) {
69 if (element->getTagName() == "header") {
70 document->getHeader() = musx::factory::HeaderFactory::create(element);
71 } else if (element->getTagName() == "options") {
72 document->getOptions() = musx::factory::OptionsFactory::create(element, document, elementLinker);
73 } else if (element->getTagName() == "others") {
74 document->getOthers() = musx::factory::OthersFactory::create(element, document, elementLinker);
75 } else if (element->getTagName() == "details") {
76 document->getDetails() = musx::factory::DetailsFactory::create(element, document, elementLinker);
77 } else if (element->getTagName() == "entries") {
78 document->getEntries() = musx::factory::EntryFactory::create(element, document, elementLinker);
79 } else if (element->getTagName() == "texts") {
80 document->getTexts() = musx::factory::TextsFactory::create(element, document, elementLinker);
81 }
82 }
83 if (!document->getHeader()) document->getHeader() = std::make_shared<musx::dom::Header>();
84 if (!document->getOptions()) document->getOptions() = std::make_shared<musx::dom::OptionsPool>();
85 if (!document->getOthers()) document->getOthers() = std::make_shared<musx::dom::OthersPool>();
86 if (!document->getDetails()) document->getDetails() = std::make_shared<musx::dom::DetailsPool>();
87 if (!document->getEntries()) document->getEntries() = std::make_shared<musx::dom::EntryPool>();
88 if (!document->getTexts()) document->getTexts() = std::make_shared<musx::dom::TextsPool>();
89
90#ifdef MUSX_DISPLAY_NODE_NAMES
91 util::Logger::log(util::Logger::LogLevel::Verbose, "============");
92#endif
93 elementLinker.resolveAll(document);
94
95 document->m_maxBlankPages = 0;
96 auto linkedParts = document->getOthers()->getArray<PartDefinition>(SCORE_PARTID);
97 for (const auto& part : linkedParts) {
98 part->numberOfLeadingBlankPages = 0;
99 auto pages = document->getOthers()->getArray<Page>(part->getCmper());
100 part->numberOfPages = int(pages.size());
101 for (const auto& page : pages) {
102 if (!page->isBlank()) {
103 break;
104 }
105 part->numberOfLeadingBlankPages++;
106 }
107 if (part->numberOfLeadingBlankPages > document->m_maxBlankPages) {
108 document->m_maxBlankPages = part->numberOfLeadingBlankPages;
109 }
110 }
111
112 return document;
113 }

125 return create<XmlDocumentType>(xmlBuffer.data(), xmlBuffer.size());
126 }
