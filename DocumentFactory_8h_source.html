133 void setEmbeddedGraphics(EmbeddedGraphicFiles&& embeddedGraphicFiles)
134 {
135 m_embeddedGraphics.clear();
136 for (auto& file : embeddedGraphicFiles) {
137 const auto parsed = parseEmbeddedGraphicFilename(file.filename);
138 if (!parsed) {
139 continue;
140 }
141 auto& entry = m_embeddedGraphics[parsed->first];
142 entry.extension = std::move(parsed->second);
143 entry.bytes = std::move(file.bytes);
144 }

148 static std::optional<std::pair<Cmper, std::string>> parseEmbeddedGraphicFilename(const std::string& filename)
149 {
150 const auto dotPos = filename.find('.');
151 if (dotPos == std::string::npos || dotPos == 0 || dotPos + 1 >= filename.size()) {
152 return std::nullopt;
153 }
154
155 const std::string_view cmperText(filename.data(), dotPos);
156 if (cmperText.empty() || !std::all_of(cmperText.begin(), cmperText.end(),
157 [](unsigned char ch) { return std::isdigit(ch) != 0; })) {
158 return std::nullopt;
159 }
160
161 unsigned long long cmperValue = 0;
162 try {
163 cmperValue = std::stoull(std::string(cmperText));
164 } catch (...) {
165 return std::nullopt;
166 }
167 if (cmperValue > static_cast<unsigned long long>((std::numeric_limits<Cmper>::max)())) {
168 return std::nullopt;
169 }
170
171 return std::make_pair(static_cast<Cmper>(cmperValue), filename.substr(dotPos + 1));
172 }

210 template <typename XmlDocumentType>
211 [[nodiscard]]
212 static DocumentPtr create(const char* data, size_t size, CreateOptions&& createOptions)
213 {
214 static_assert(std::is_base_of<musx::xml::IXmlDocument, XmlDocumentType>::value,
215 "XmlReaderType must derive from IXmlDocument.");
216
217 std::unique_ptr<musx::xml::IXmlDocument> xmlDocument = std::make_unique<XmlDocumentType>();
218 xmlDocument->loadFromBuffer(data, size);
219
220 auto rootElement = xmlDocument->getRootElement();
221 if (!rootElement || rootElement->getTagName() != "finale") {
222 throw std::invalid_argument("Missing <finale> element.");
223 }
224
225 DocumentPtr document(new Document);
226 document->m_self = document;
227 document->m_partVoicingPolicy = createOptions.partVoicingPolicy;
228 document->m_scoreDurationSeconds = parseScoreDurationSeconds<XmlDocumentType>(createOptions.getNotationMetadata());
229 document->m_embeddedGraphics = createOptions.takeEmbeddedGraphics();
230
231 ElementLinker elementLinker;
232 for (auto element = rootElement->getFirstChildElement(); element; element = element->getNextSibling()) {
233 if (element->getTagName() == "header") {
234 document->getHeader() = musx::factory::HeaderFactory::create(element);
235 } else if (element->getTagName() == "options") {
236 document->getOptions() = musx::factory::OptionsFactory::create(element, document, elementLinker);
237 } else if (element->getTagName() == "others") {
238 document->getOthers() = musx::factory::OthersFactory::create(element, document, elementLinker);
239 } else if (element->getTagName() == "details") {
240 document->getDetails() = musx::factory::DetailsFactory::create(element, document, elementLinker);
241 } else if (element->getTagName() == "entries") {
242 document->getEntries() = musx::factory::EntryFactory::create(element, document, elementLinker);
243 } else if (element->getTagName() == "texts") {
244 document->getTexts() = musx::factory::TextsFactory::create(element, document, elementLinker);
245 }
246 }
247 if (!document->getHeader()) document->getHeader() = std::make_shared<musx::dom::Header>();
248 if (!document->getOptions()) document->getOptions() = std::make_shared<musx::dom::OptionsPool>(document);
249 if (!document->getOthers()) document->getOthers() = std::make_shared<musx::dom::OthersPool>(document);
250 if (!document->getDetails()) document->getDetails() = std::make_shared<musx::dom::DetailsPool>(document);
251 if (!document->getEntries()) document->getEntries() = std::make_shared<musx::dom::EntryPool>(document);
252 if (!document->getTexts()) document->getTexts() = std::make_shared<musx::dom::TextsPool>(document);
253
254#ifdef MUSX_DISPLAY_NODE_NAMES
255 util::Logger::log(util::Logger::LogLevel::Verbose, "============");
256#endif
257 elementLinker.resolveAll(document);
258 document->m_instruments = document->createInstrumentMap(SCORE_PARTID);
259
260 document->m_maxBlankPages = 0;
261 auto linkedParts = document->getOthers()->getArray<PartDefinition>(SCORE_PARTID);
262 for (const auto& part : linkedParts) {
263 auto mutablePart = const_cast<PartDefinition*>(part.get());
264 mutablePart->numberOfLeadingBlankPages = 0;
265 auto pages = document->getOthers()->getArray<Page>(part->getCmper());
266 mutablePart->numberOfPages = int(pages.size());
267 for (const auto& page : pages) {
268 if (!page->isBlank()) {
269 break;
270 }
271 mutablePart->numberOfLeadingBlankPages++;
272 }
273 if (mutablePart->numberOfLeadingBlankPages > document->m_maxBlankPages) {
274 document->m_maxBlankPages = part->numberOfLeadingBlankPages;
275 }
276 }
277
278 return document;
279 }

296 template <typename XmlDocumentType>
297 static std::optional<double> parseScoreDurationSeconds(const std::vector<char>& notationMetadata)
298 {
299 if (notationMetadata.empty()) {
300 return std::nullopt;
301 }
302
303 std::unique_ptr<musx::xml::IXmlDocument> xmlDocument = std::make_unique<XmlDocumentType>();
304 try {
305 xmlDocument->loadFromBuffer(notationMetadata.data(), notationMetadata.size());
306 } catch (...) {
307 return std::nullopt;
308 }
309
310 auto rootElement = xmlDocument->getRootElement();
311 if (!rootElement || rootElement->getTagName() != "metadata") {
312 return std::nullopt;
313 }
314 if (auto fileInfo = rootElement->getFirstChildElement("fileInfo")) {
315 auto creatorString = fileInfo->getFirstChildElement("creatorString");
316 if (!creatorString || creatorString->getTextTrimmed().rfind("Finale", 0) != 0) {
317 return std::nullopt;
318 }
319 if (auto scoreDuration = fileInfo->getFirstChildElement("scoreDuration")) {
320 try {
321 return scoreDuration->getTextAs<double>();
322 } catch (...) {
323 return std::nullopt;
324 }
325 }
326 }
327 return std::nullopt;
328 }
