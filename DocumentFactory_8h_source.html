123 template <typename XmlDocumentType>
124 [[nodiscard]]
125 static DocumentPtr create(const char* data, size_t size, const CreateOptions& createOptions)
126 {
127 static_assert(std::is_base_of<musx::xml::IXmlDocument, XmlDocumentType>::value,
128 "XmlReaderType must derive from IXmlDocument.");
129
130 std::unique_ptr<musx::xml::IXmlDocument> xmlDocument = std::make_unique<XmlDocumentType>();
131 xmlDocument->loadFromBuffer(data, size);
132
133 auto rootElement = xmlDocument->getRootElement();
134 if (!rootElement || rootElement->getTagName() != "finale") {
135 throw std::invalid_argument("Missing <finale> element.");
136 }
137
138 DocumentPtr document(new Document);
139 document->m_self = document;
140 document->m_partVoicingPolicy = createOptions.partVoicingPolicy;
141 document->m_scoreDurationSeconds = parseScoreDurationSeconds<XmlDocumentType>(createOptions.getNotationMetadata());
142
143 ElementLinker elementLinker;
144 for (auto element = rootElement->getFirstChildElement(); element; element = element->getNextSibling()) {
145 if (element->getTagName() == "header") {
146 document->getHeader() = musx::factory::HeaderFactory::create(element);
147 } else if (element->getTagName() == "options") {
148 document->getOptions() = musx::factory::OptionsFactory::create(element, document, elementLinker);
149 } else if (element->getTagName() == "others") {
150 document->getOthers() = musx::factory::OthersFactory::create(element, document, elementLinker);
151 } else if (element->getTagName() == "details") {
152 document->getDetails() = musx::factory::DetailsFactory::create(element, document, elementLinker);
153 } else if (element->getTagName() == "entries") {
154 document->getEntries() = musx::factory::EntryFactory::create(element, document, elementLinker);
155 } else if (element->getTagName() == "texts") {
156 document->getTexts() = musx::factory::TextsFactory::create(element, document, elementLinker);
157 }
158 }
159 if (!document->getHeader()) document->getHeader() = std::make_shared<musx::dom::Header>();
160 if (!document->getOptions()) document->getOptions() = std::make_shared<musx::dom::OptionsPool>(document);
161 if (!document->getOthers()) document->getOthers() = std::make_shared<musx::dom::OthersPool>(document);
162 if (!document->getDetails()) document->getDetails() = std::make_shared<musx::dom::DetailsPool>(document);
163 if (!document->getEntries()) document->getEntries() = std::make_shared<musx::dom::EntryPool>(document);
164 if (!document->getTexts()) document->getTexts() = std::make_shared<musx::dom::TextsPool>(document);
165
166#ifdef MUSX_DISPLAY_NODE_NAMES
167 util::Logger::log(util::Logger::LogLevel::Verbose, "============");
168#endif
169 elementLinker.resolveAll(document);
170 document->m_instruments = document->createInstrumentMap(SCORE_PARTID);
171
172 document->m_maxBlankPages = 0;
173 auto linkedParts = document->getOthers()->getArray<PartDefinition>(SCORE_PARTID);
174 for (const auto& part : linkedParts) {
175 auto mutablePart = const_cast<PartDefinition*>(part.get());
176 mutablePart->numberOfLeadingBlankPages = 0;
177 auto pages = document->getOthers()->getArray<Page>(part->getCmper());
178 mutablePart->numberOfPages = int(pages.size());
179 for (const auto& page : pages) {
180 if (!page->isBlank()) {
181 break;
182 }
183 mutablePart->numberOfLeadingBlankPages++;
184 }
185 if (mutablePart->numberOfLeadingBlankPages > document->m_maxBlankPages) {
186 document->m_maxBlankPages = part->numberOfLeadingBlankPages;
187 }
188 }
189
190 return document;
191 }

208 template <typename XmlDocumentType>
209 static std::optional<double> parseScoreDurationSeconds(const std::vector<char>& notationMetadata)
210 {
211 if (notationMetadata.empty()) {
212 return std::nullopt;
213 }
214
215 std::unique_ptr<musx::xml::IXmlDocument> xmlDocument = std::make_unique<XmlDocumentType>();
216 try {
217 xmlDocument->loadFromBuffer(notationMetadata.data(), notationMetadata.size());
218 } catch (...) {
219 return std::nullopt;
220 }
221
222 auto rootElement = xmlDocument->getRootElement();
223 if (!rootElement || rootElement->getTagName() != "metadata") {
224 return std::nullopt;
225 }
226 if (auto fileInfo = rootElement->getFirstChildElement("fileInfo")) {
227 auto creatorString = fileInfo->getFirstChildElement("creatorString");
228 if (!creatorString || creatorString->getTextTrimmed().rfind("Finale", 0) != 0) {
229 return std::nullopt;
230 }
231 if (auto scoreDuration = fileInfo->getFirstChildElement("scoreDuration")) {
232 try {
233 return scoreDuration->getTextAs<double>();
234 } catch (...) {
235 return std::nullopt;
236 }
237 }
238 }
239 return std::nullopt;
240 }
